26 void pixel2metric_facedown(double x_e, double y_e, double alt, double theta, double theta_0, vector<double> obj_pix, vector<double> &O_m)
27 {
28  // find vector from middle of camera
29  double r_p[2] = {0, 0};
30  r_p[0] = obj_pix[0] - PIXLES[0];
31  r_p[1] = obj_pix[1] - PIXLES[1];
32 
33  double O_mx = r_p[0]/3779.527;
34  double O_my = (-1.0)*r_p[1]/3779.527;
35 
36  // convert degrees to radians
37  theta = theta * (PI/180.0);
38  theta_0 = theta_0 * (PI/180.0);
39 
40  // compute coefficients
41  double c11 = cos(theta_0)*cos(theta) + sin(theta)*sin(theta_0);
42  double c12 = sin(theta)*cos(theta_0) - sin(theta_0)*cos(theta);
43  double c21 = sin(theta_0)*cos(theta) - sin(theta)*cos(theta_0);
44  double c22 = sin(theta)*sin(theta_0) + cos(theta)*cos(theta_0);
45 
46  // Transform drone pos to ENU
47  double x_d = x_e*cos(theta_0) - y_e*sin(theta_0);
48  double y_d = x_e*sin(theta_0) + y_e*cos(theta_0);
49 
50  // transform roomba position from drone to gym frame
51  double r_x = O_mx*c11 + O_my*c12 + x_d;
52  double r_y = O_mx*c21 + O_my*c22 + y_d;
53 
54  // O_m[0] = O_mx;
55  // O_m[1] = O_my;
56  ROS_INFO("transformed to x: %f y: %f z: %f meters \n", r_x, r_y, rngfnd.range);
57  roombaPose.pose.position.x = O_my;
58  roombaPose.pose.position.y = O_mx;
59  roombaPose.pose.position.z = rngfnd.range;
60 }
