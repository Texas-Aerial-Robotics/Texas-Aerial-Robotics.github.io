127  void state_quat_cb(const mavros_msgs::HilStateQuaternion::ConstPtr &req) {
128  mavlink::common::msg::HIL_STATE_QUATERNION state_quat;
129 
130  state_quat.time_usec = req->header.stamp.toNSec() / 1000;
131  auto q = ftf::transform_orientation_baselink_aircraft(
132  ftf::transform_orientation_enu_ned(
133  ftf::to_eigen(req->orientation)));
134  ftf::quaternion_to_mavlink(q, state_quat.attitude_quaternion);
135  state_quat.lat = req->geo.latitude * 1E7;
136  state_quat.lon = req->geo.longitude * 1E7;
137  // @warning geographic_msgs/GeoPoint.msg uses WGS 84 reference ellipsoid
138  // @TODO: Convert altitude to AMSL to be received by the FCU
139  // related to issue #529
140  state_quat.alt = req->geo.altitude * 1E3;
141  state_quat.ind_airspeed = req->ind_airspeed * 1E2;
142  state_quat.true_airspeed = req->true_airspeed * 1E2;
143  // WRT world frame
144  auto ang_vel = ftf::transform_frame_enu_ned(
145  ftf::transform_frame_baselink_aircraft(
146  ftf::to_eigen(req->angular_velocity)));
147  auto lin_vel = ftf::transform_frame_enu_ned<Eigen::Vector3d>(
148  ftf::to_eigen(req->linear_velocity)) * 1E2;
149  // linear acceleration - WRT world frame
150  auto lin_acc = ftf::transform_frame_baselink_aircraft(
151  ftf::to_eigen(req->linear_acceleration));
152 
153  // [[[cog:
154  // for a, b in zip(('rollspeed', 'pitchspeed', 'yawspeed'), "xyz"):
155  // cog.outl("state_quat.%s = ang_vel.%s();" % (a, b))
156  // for f in "xyz":
157  // cog.outl("state_quat.v%s = lin_vel.%s();" % (f, f))
158  // for f in "xyz":
159  // cog.outl("state_quat.%sacc = lin_acc.%s();" % (f, f))
160  // ]]]
161  state_quat.rollspeed = ang_vel.x();
162  state_quat.pitchspeed = ang_vel.y();
163  state_quat.yawspeed = ang_vel.z();
164  state_quat.vx = lin_vel.x();
165  state_quat.vy = lin_vel.y();
166  state_quat.vz = lin_vel.z();
167  state_quat.xacc = lin_acc.x();
168  state_quat.yacc = lin_acc.y();
169  state_quat.zacc = lin_acc.z();
