174  offset += f.array_length * mapType[f.type][1]
175  continue
176

179  offset+=1
180  elif (f.type == 'int8_t'):
181  outf.write("\t\t\tbytes[offset + %s] = unchecked((byte)msg.%s);\n" % (offset,mapFieldName.get(f.name, f.name)))
182  offset+=1
183  elif (f.type == 'char'):
