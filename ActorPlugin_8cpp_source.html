35 void ActorPlugin::Load(physics::ModelPtr _model, sdf::ElementPtr _sdf)
36 {
37  this->sdf = _sdf;
38  this->actor = boost::dynamic_pointer_cast<physics::Actor>(_model);
39  this->world = this->actor->GetWorld();
40 
41  this->connections.push_back(event::Events::ConnectWorldUpdateBegin(
42  std::bind(&ActorPlugin::OnUpdate, this, std::placeholders::_1)));
43 
44  this->Reset();
45 
46  // Read in the target weight
47  if (_sdf->HasElement("target_weight"))
48  this->targetWeight = _sdf->Get<double>("target_weight");
49  else
50  this->targetWeight = 1.15;
51 
52  // Read in the obstacle weight
53  if (_sdf->HasElement("obstacle_weight"))
54  this->obstacleWeight = _sdf->Get<double>("obstacle_weight");
55  else
56  this->obstacleWeight = 1.5;
57 
58  // Read in the animation factor (applied in the OnUpdate function).
59  if (_sdf->HasElement("animation_factor"))
60  this->animationFactor = _sdf->Get<double>("animation_factor");
61  else
62  this->animationFactor = 4.5;
63 
64  // Add our own name to models we should ignore when avoiding obstacles.
65  this->ignoreModels.push_back(this->actor->GetName());
66 
67  // Read in the other obstacles to ignore
68  if (_sdf->HasElement("ignore_obstacles"))
69  {
70  sdf::ElementPtr modelElem =
71  _sdf->GetElement("ignore_obstacles")->GetElement("model");
72  while (modelElem)
73  {
74  this->ignoreModels.push_back(modelElem->Get<std::string>());
75  modelElem = modelElem->GetNextElement("model");
76  }
77  }
78 }

81 void ActorPlugin::Reset()
82 {
83  this->velocity = 0.8;
84  this->lastUpdate = 0;
85 
86  if (this->sdf && this->sdf->HasElement("target"))
87  this->target = this->sdf->Get<ignition::math::Vector3d>("target");
88  else
89  this->target = ignition::math::Vector3d(0, -5, 1.2138);
90 
91  auto skelAnims = this->actor->SkeletonAnimations();
92  if (skelAnims.find(WALKING_ANIMATION) == skelAnims.end())
93  {
94  gzerr << "Skeleton animation " << WALKING_ANIMATION << " not found.\n";
95  }
96  else
97  {
98  // Create custom trajectory
99  this->trajectoryInfo.reset(new physics::TrajectoryInfo());
100  this->trajectoryInfo->type = WALKING_ANIMATION;
101  this->trajectoryInfo->duration = 1.0;
102 
103  this->actor->SetCustomTrajectory(this->trajectoryInfo);
104  }
105 }

108 void ActorPlugin::ChooseNewTarget()
109 {
110  ignition::math::Vector3d newTarget(this->target);
111  while ((newTarget - this->target).Length() < 2.0)
112  {
113  newTarget.X(ignition::math::Rand::DblUniform(-3, 3.5));
114  newTarget.Y(ignition::math::Rand::DblUniform(-10, 2));
115 
116  for (unsigned int i = 0; i < this->world->ModelCount(); ++i)
117  {
118  double dist = (this->world->ModelByIndex(i)->WorldPose().Pos()
119  - newTarget).Length();
120  if (dist < 2.0)
121  {
122  newTarget = this->target;
123  break;
124  }
125  }
126  }
127  this->target = newTarget;
128 }

131 void ActorPlugin::HandleObstacles(ignition::math::Vector3d &_pos)
132 {
133  for (unsigned int i = 0; i < this->world->ModelCount(); ++i)
134  {
135  physics::ModelPtr model = this->world->ModelByIndex(i);
136  if (std::find(this->ignoreModels.begin(), this->ignoreModels.end(),
137  model->GetName()) == this->ignoreModels.end())
138  {
139  ignition::math::Vector3d offset = model->WorldPose().Pos() -
140  this->actor->WorldPose().Pos();
141  double modelDist = offset.Length();
142  if (modelDist < 4.0)
143  {
144  double invModelDist = this->obstacleWeight / modelDist;
145  offset.Normalize();
146  offset *= invModelDist;
147  _pos -= offset;
148  }
149  }
150  }
151 }

154 void ActorPlugin::OnUpdate(const common::UpdateInfo &_info)
155 {
156  // Time delta
157  double dt = (_info.simTime - this->lastUpdate).Double();
158 
159  ignition::math::Pose3d pose = this->actor->WorldPose();
160  ignition::math::Vector3d pos = this->target - pose.Pos();
161  ignition::math::Vector3d rpy = pose.Rot().Euler();
162 
163  double distance = pos.Length();
164 
165  // Choose a new target position if the actor has reached its current
166  // target.
167  if (distance < 0.3)
168  {
169  this->ChooseNewTarget();
170  pos = this->target - pose.Pos();
171  }
172 
173  // Normalize the direction vector, and apply the target weight
174  pos = pos.Normalize() * this->targetWeight;
175 
176  // Adjust the direction vector by avoiding obstacles
177  this->HandleObstacles(pos);
178 
179  // Compute the yaw orientation
180  ignition::math::Angle yaw = atan2(pos.Y(), pos.X()) + 1.5707 - rpy.Z();
181  yaw.Normalize();
182 
183  // Rotate in place, instead of jumping.
184  if (std::abs(yaw.Radian()) > IGN_DTOR(10))
185  {
186  pose.Rot() = ignition::math::Quaterniond(1.5707, 0, rpy.Z()+
187  yaw.Radian()*0.001);
188  }
189  else
190  {
191  pose.Pos() += pos * this->velocity * dt;
192  pose.Rot() = ignition::math::Quaterniond(1.5707, 0, rpy.Z()+yaw.Radian());
193  }
194 
195  // Make sure the actor stays within bounds
196  pose.Pos().X(std::max(-3.0, std::min(3.5, pose.Pos().X())));
197  pose.Pos().Y(std::max(-10.0, std::min(2.0, pose.Pos().Y())));
198  pose.Pos().Z(1.2138);
199 
200  // Distance traveled is used to coordinate motion with the walking
201  // animation
202  double distanceTraveled = (pose.Pos() -
203  this->actor->WorldPose().Pos()).Length();
204 
205  this->actor->SetWorldPose(pose, false, false);
206  this->actor->SetScriptTime(this->actor->ScriptTime() +
207  (distanceTraveled * this->animationFactor));
208  this->lastUpdate = _info.simTime;
209 }
