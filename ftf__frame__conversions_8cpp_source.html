66 Eigen::Quaterniond transform_orientation(const Eigen::Quaterniond &q, const StaticTF transform)
67 {
68  // Transform the attitude representation from frame to frame.
69  // The proof for this transform can be seen
70  // http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/transforms/
71  switch (transform) {
72  case StaticTF::NED_TO_ENU:
73  case StaticTF::ENU_TO_NED:
74  return NED_ENU_Q * q;
75 
76  case StaticTF::AIRCRAFT_TO_BASELINK:
77  case StaticTF::BASELINK_TO_AIRCRAFT:
78  return q * AIRCRAFT_BASELINK_Q;
79  }
80 }

114 Covariance6d transform_static_frame(const Covariance6d &cov, const StaticTF transform)
115 {
116  Covariance6d cov_out_;
117  Matrix6d R = Matrix6d::Zero(); // not `auto` because Zero ret is const
118 
119  EigenMapConstCovariance6d cov_in(cov.data());
120  EigenMapCovariance6d cov_out(cov_out_.data());
121 
122  switch (transform) {
123  case StaticTF::NED_TO_ENU:
124  case StaticTF::ENU_TO_NED:
125  R.block<3, 3>(0, 0) =
126  R.block<3, 3>(3, 3) = NED_ENU_R;
127 
128  cov_out = R * cov_in * R.transpose();
129  return cov_out_;
130 
131  case StaticTF::AIRCRAFT_TO_BASELINK:

133  R.block<3, 3>(0, 0) =
134  R.block<3, 3>(3, 3) = AIRCRAFT_BASELINK_R;

170 Eigen::Vector3d transform_static_frame(const Eigen::Vector3d &vec, const Eigen::Vector3d &map_origin, const StaticTF transform)
171 {
173  static constexpr double DEG_TO_RAD = (M_PI / 180.0);
174 
175  // Don't forget to convert from degrees to radians
176  const double sin_lat = std::sin(map_origin.x() * DEG_TO_RAD);
177  const double sin_lon = std::sin(map_origin.y() * DEG_TO_RAD);
178  const double cos_lat = std::cos(map_origin.x() * DEG_TO_RAD);
179  const double cos_lon = std::cos(map_origin.y() * DEG_TO_RAD);
180 
195  Eigen::Matrix3d R;
196  R << -sin_lon, cos_lon, 0.0,
197  -cos_lon * sin_lat, -sin_lon * sin_lat, cos_lat,
198  cos_lon * cos_lat, sin_lon * cos_lat, sin_lat;
199 
200  switch (transform) {
201  case StaticTF::ECEF_TO_ENU:
202  return R * vec;
203 
204  case StaticTF::ENU_TO_ECEF:
205  // ENU to ECEF rotation is just an inverse rotation from ECEF to ENU, which means transpose.
206  R.transposeInPlace();
207  return R * vec;
208  }
209 }
