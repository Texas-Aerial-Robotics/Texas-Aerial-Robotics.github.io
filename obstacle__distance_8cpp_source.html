63  void obstacle_cb(const sensor_msgs::LaserScan::ConstPtr &req)
64  {
65  mavlink::common::msg::OBSTACLE_DISTANCE obstacle {};
66 
67  auto n = std::min(req->ranges.size(), obstacle.distances.size());
68  auto cm_ranges = Eigen::Map<const Eigen::VectorXf>(req->ranges.data(), n) * 1e2;
69  Eigen::Map<Eigen::Matrix<uint16_t, Eigen::Dynamic, 1> > map_distances(obstacle.distances.data(), n);
70 
71  obstacle.time_usec = req->header.stamp.toNSec() / 1000;
72  obstacle.sensor_type = utils::enum_value(MAV_DISTANCE_SENSOR::LASER);
73  map_distances = cm_ranges.cast<uint16_t>();
74  std::fill(obstacle.distances.begin() + n, obstacle.distances.end(), UINT16_MAX);
75  obstacle.increment = req->angle_increment * RAD_TO_DEG;
76  obstacle.min_distance = req->range_min * 1e2;
77  obstacle.max_distance = req->range_max * 1e2;
78 
79  ROS_DEBUG_STREAM_NAMED("obstacle_distance", "OBSDIST: sensor type: " << utils::to_string_enum<MAV_DISTANCE_SENSOR>(obstacle.sensor_type)
80  << std::endl << obstacle.to_yaml());
81 
82  UAS_FCU(m_uas)->send_message_ignore_drop(obstacle);
