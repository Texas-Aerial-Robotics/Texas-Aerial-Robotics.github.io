67  std::vector<float> data;

70  static constexpr size_t ACC_SIZE = 50;
71 
75  float calculate_variance(float range) {
76  if (data.size() < ACC_SIZE) {
77  // limits the size of the array to 50 elements
78  data.reserve(ACC_SIZE);
79  data.push_back(range);
80  }
81  else {
82  data[data_index] = range; // it starts rewriting the values from 1st element
83  if (++data_index > ACC_SIZE - 1)
84  data_index = 0; // restarts the index when achieves the last element
85  }
86 
87  float average, variance, sum = 0, sum_ = 0;
88 
89  /* Compute the sum of all elements */
90  for (auto d : data)
91  sum += d;
92 
93  average = sum / data.size();
94 
95  /* Compute the variance */
96  for (auto d : data)
97  sum_ += (d - average) * (d - average);
98 
99  variance = sum_ / data.size();
100 
101  return variance;
102  }

197  void handle_distance_sensor(const mavlink::mavlink_message_t *msg, mavlink::common::msg::DISTANCE_SENSOR &dist_sen)
198  {
199  using mavlink::common::MAV_SENSOR_ORIENTATION;
200  using mavlink::common::MAV_DISTANCE_SENSOR;
201 
202  auto it = sensor_map.find(dist_sen.id);
203  if (it == sensor_map.end()) {
204  ROS_ERROR_NAMED("distance_sensor",
205  "DS: no mapping for sensor id: %d, type: %d, orientation: %d",
206  dist_sen.id, dist_sen.type, dist_sen.orientation);
207  return;
208  }
209 
210  auto sensor = it->second;
211  if (sensor->is_subscriber) {
212  ROS_ERROR_ONCE_NAMED("distance_sensor",
213  "DS: %s (id %d) is subscriber, but i got sensor data for that id from FCU",
214  sensor->topic_name.c_str(), sensor->sensor_id);
215 
216  return;
217  }
218 
219  if (sensor->orientation >= 0 && dist_sen.orientation != sensor->orientation) {
220  ROS_ERROR_NAMED("distance_sensor",
221  "DS: %s: received sensor data has different orientation (%s) than in config (%s)!",
222  sensor->topic_name.c_str(),
223  utils::to_string_enum<MAV_SENSOR_ORIENTATION>(dist_sen.orientation).c_str(),
224  utils::to_string_enum<MAV_SENSOR_ORIENTATION>(sensor->orientation).c_str());
225  }
226 
227  auto range = boost::make_shared<sensor_msgs::Range>();
228 
229  range->header = m_uas->synchronized_header(sensor->frame_id, dist_sen.time_boot_ms);
230 
231  range->min_range = dist_sen.min_distance * 1E-2;// in meters
232  range->max_range = dist_sen.max_distance * 1E-2;
233  range->field_of_view = sensor->field_of_view;
234 
235  switch (dist_sen.type) {
236  case enum_value(MAV_DISTANCE_SENSOR::LASER):
237  case enum_value(MAV_DISTANCE_SENSOR::RADAR):
238  case enum_value(MAV_DISTANCE_SENSOR::UNKNOWN):
239  range->radiation_type = sensor_msgs::Range::INFRARED;
240  break;
241  case enum_value(MAV_DISTANCE_SENSOR::ULTRASOUND):
242  range->radiation_type = sensor_msgs::Range::ULTRASOUND;
243  break;
244  default:
245  ROS_ERROR_NAMED("distance_sensor",
246  "DS: %s: Wrong/undefined type of sensor (type: %d). Droping!...",
247  sensor->topic_name.c_str(), dist_sen.type);
248  return;
249  }
250 
251  range->range = dist_sen.current_distance * 1E-2; // in meters
252 
253  if (sensor->send_tf) {
254  /* variables init */
255  auto q = utils::sensor_orientation_matching(static_cast<MAV_SENSOR_ORIENTATION>(dist_sen.orientation));
256 
257  geometry_msgs::TransformStamped transform;
258 
259  transform.header = m_uas->synchronized_header(base_frame_id, dist_sen.time_boot_ms);
260  transform.child_frame_id = sensor->frame_id;
261 
262  /* rotation and position set */
263  tf::quaternionEigenToMsg(q, transform.transform.rotation);
264  tf::vectorEigenToMsg(sensor->position, transform.transform.translation);
265 
266  /* transform broadcast */
267  m_uas->tf2_broadcaster.sendTransform(transform);
268  }
269 
270  sensor->pub.publish(range);
271  }

274 void DistanceSensorItem::range_cb(const sensor_msgs::Range::ConstPtr &msg)
275 {
276  using mavlink::common::MAV_DISTANCE_SENSOR;
277 
278  uint8_t type = 0;
279  uint8_t covariance_ = 0;
280 
281  if (covariance > 0) covariance_ = covariance;
282  else covariance_ = uint8_t(calculate_variance(msg->range) * 1E2); // in cm
283 
285  ROS_DEBUG_NAMED("distance_sensor", "DS: %d: sensor variance: %f", sensor_id, calculate_variance(msg->range) * 1E2);
286 
287  // current mapping, may change later
288  if (msg->radiation_type == sensor_msgs::Range::INFRARED)
289  type = enum_value(MAV_DISTANCE_SENSOR::LASER);
290  else if (msg->radiation_type == sensor_msgs::Range::ULTRASOUND)
291  type = enum_value(MAV_DISTANCE_SENSOR::ULTRASOUND);
292 
293  owner->distance_sensor(
294  msg->header.stamp.toNSec() / 1000000,
295  msg->min_range / 1E-2,
296  msg->max_range / 1E-2,
297  msg->range / 1E-2,
298  type,
299  sensor_id,
300  orientation,
301  covariance_);
302 }

304 DistanceSensorItem::Ptr DistanceSensorItem::create_item(DistanceSensorPlugin *owner, std::string topic_name)
305 {
306  auto p = boost::make_shared<DistanceSensorItem>();
307  std::string orientation_str;
308 
309  ros::NodeHandle pnh(owner->dist_nh, topic_name);
310 
311  p->owner = owner;
312  p->topic_name = topic_name;
313 
314  // load and parse params
315  // first decide the type of topic (sub or pub)
316  pnh.param("subscriber", p->is_subscriber, false);
317 
318  // sensor id
319  int id;
320  if (!pnh.getParam("id", id)) {
321  ROS_ERROR_NAMED("distance_sensor", "DS: %s: `id` not set!", topic_name.c_str());
322  p.reset(); return p; // return nullptr because of a bug related to gcc 4.6
323  }
324  p->sensor_id = id;
325 
326  // orientation, checks later
327  if (!pnh.getParam("orientation", orientation_str))
328  p->orientation = -1; // not set
329  else
330  // lookup for numeric value
331  p->orientation = utils::sensor_orientation_from_str(orientation_str);
332 
333 
334  if (!p->is_subscriber) {
335  // publisher params
336  // frame_id and FOV is required
337  if (!pnh.getParam("frame_id", p->frame_id)) {
338  ROS_ERROR_NAMED("distance_sensor", "DS: %s: `frame_id` not set!", topic_name.c_str());
339  p.reset(); return p; // nullptr
340  }
341 
342  if (!pnh.getParam("field_of_view", p->field_of_view)) {
343  ROS_ERROR_NAMED("distance_sensor", "DS: %s: sensor FOV not set!", topic_name.c_str());
344  p.reset(); return p; // nullptr
345  }
346 
347  // unset allowed, setted wrong - not
348  if (p->orientation == -1 && !orientation_str.empty()) {
349  ROS_ERROR_NAMED("distance_sensor", "DS: %s: defined orientation (%s) is not valid!",
350  topic_name.c_str(), orientation_str.c_str());
351  p.reset(); return p; // nullptr
352  }
353 
354  // optional
355  pnh.param("send_tf", p->send_tf, false);
356  if (p->send_tf) { // sensor position defined if 'send_tf' set to TRUE
357  pnh.param("sensor_position/x", p->position.x(), 0.0);
358  pnh.param("sensor_position/y", p->position.y(), 0.0);
359  pnh.param("sensor_position/z", p->position.z(), 0.0);
360  ROS_DEBUG_NAMED("sensor_position", "DS: %s: Sensor position at: %f, %f, %f", topic_name.c_str(),
361  p->position.x(), p->position.y(), p->position.z());
362  }
363  }
364  else {
365  // subscriber params
366  // orientation is required
367  if (orientation_str.empty()) {
368  ROS_ERROR_NAMED("distance_sensor", "DS: %s: orientation not set!", topic_name.c_str());
369  p.reset(); return p; // nullptr
370  }
371  else if (p->orientation == -1) {
372  ROS_ERROR_NAMED("distance_sensor", "DS: %s: defined orientation (%s) is not valid!", topic_name.c_str(), orientation_str.c_str());
373  p.reset(); return p; // nullptr
374  }
375 
376  // optional
377  pnh.param("covariance", p->covariance, 0);
378  }
379 
380  // create topic handles
381  if (!p->is_subscriber)
382  p->pub = owner->dist_nh.advertise<sensor_msgs::Range>(topic_name, 10);
383  else
384  p->sub = owner->dist_nh.subscribe(topic_name, 10, &DistanceSensorItem::range_cb, p.get());
385 
386  return p;
387 }
