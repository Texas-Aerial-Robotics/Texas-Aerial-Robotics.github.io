135  void setup_covariance(ftf::Covariance3d &cov, double stdev)
136  {
137  ftf::EigenMapCovariance3d c(cov.data());
138 
139  c.setZero();
140  if (stdev) {
141  double sr = stdev * stdev;
142  c.diagonal() << sr, sr, sr;
143  }
144  else {
145  c(0,0) = -1.0;
146  }
147  }

157  void publish_imu_data(uint32_t time_boot_ms, Eigen::Quaterniond &orientation_enu,
158  Eigen::Quaterniond &orientation_ned, Eigen::Vector3d &gyro_flu, Eigen::Vector3d &gyro_frd)
159  {
160  auto imu_ned_msg = boost::make_shared<sensor_msgs::Imu>();
161  auto imu_enu_msg = boost::make_shared<sensor_msgs::Imu>();
162 
163  // Fill message header
164  imu_enu_msg->header = m_uas->synchronized_header(frame_id, time_boot_ms);
165  imu_ned_msg->header = m_uas->synchronized_header("aircraft", time_boot_ms);
166 
167  // Convert from Eigen::Quaternond to geometry_msgs::Quaternion
168  tf::quaternionEigenToMsg(orientation_enu, imu_enu_msg->orientation);
169  tf::quaternionEigenToMsg(orientation_ned, imu_ned_msg->orientation);
170 
171  // Convert from Eigen::Vector3d to geometry_msgs::Vector3
172  tf::vectorEigenToMsg(gyro_flu, imu_enu_msg->angular_velocity);
173  tf::vectorEigenToMsg(gyro_frd, imu_ned_msg->angular_velocity);
174 
175  // Eigen::Vector3d from HIGHRES_IMU or RAW_IMU, to geometry_msgs::Vector3
176  tf::vectorEigenToMsg(linear_accel_vec_flu, imu_enu_msg->linear_acceleration);
177  tf::vectorEigenToMsg(linear_accel_vec_frd, imu_ned_msg->linear_acceleration);
178 
179  // Pass ENU msg covariances
180  imu_enu_msg->orientation_covariance = orientation_cov;
181  imu_enu_msg->angular_velocity_covariance = angular_velocity_cov;
182  imu_enu_msg->linear_acceleration_covariance = linear_acceleration_cov;
183 
184  // Pass NED msg covariances
185  imu_ned_msg->orientation_covariance = orientation_cov;
186  imu_ned_msg->angular_velocity_covariance = angular_velocity_cov;
187  imu_ned_msg->linear_acceleration_covariance = linear_acceleration_cov;
188 
192  // [store_enu]
193  m_uas->update_attitude_imu_enu(imu_enu_msg);
194  // [store_enu]
195 
199  // [store_enu]
200  m_uas->update_attitude_imu_ned(imu_ned_msg);
201  // [store_ned]
202 
206  // [pub_enu]
207  imu_pub.publish(imu_enu_msg);
208  // [pub_enu]
209  }

268  void handle_attitude(const mavlink::mavlink_message_t *msg, mavlink::common::msg::ATTITUDE &att)
269  {
270  if (has_att_quat)
271  return;
272 
276  // [ned_aircraft_orient1]
277  auto ned_aircraft_orientation = ftf::quaternion_from_rpy(att.roll, att.pitch, att.yaw);
278  // [ned_aircraft_orient1]
279 
283  // [frd_ang_vel1]
284  auto gyro_frd = Eigen::Vector3d(att.rollspeed, att.pitchspeed, att.yawspeed);
285  // [frd_ang_vel1]
286 
291  // [ned->baselink->enu]
292  auto enu_baselink_orientation = ftf::transform_orientation_aircraft_baselink(
293  ftf::transform_orientation_ned_enu(ned_aircraft_orientation));
294  // [ned->baselink->enu]
295 
300  // [rotate_gyro]
301  auto gyro_flu = ftf::transform_frame_aircraft_baselink(gyro_frd);
302  // [rotate_gyro]
303 
304  publish_imu_data(att.time_boot_ms, enu_baselink_orientation, ned_aircraft_orientation, gyro_flu, gyro_frd);
305  }

355  void handle_highres_imu(const mavlink::mavlink_message_t *msg, mavlink::common::msg::HIGHRES_IMU &imu_hr)
356  {
357  ROS_INFO_COND_NAMED(!has_hr_imu, "imu", "IMU: High resolution IMU detected!");
358  has_hr_imu = true;
359 
360  auto header = m_uas->synchronized_header(frame_id, imu_hr.time_usec);
368  // [accel_available]
369  if (imu_hr.fields_updated & ((7 << 3) | (7 << 0))) {
370  auto gyro_flu = ftf::transform_frame_aircraft_baselink(Eigen::Vector3d(imu_hr.xgyro, imu_hr.ygyro, imu_hr.zgyro));
371 
372  auto accel_frd = Eigen::Vector3d(imu_hr.xacc, imu_hr.yacc, imu_hr.zacc);
373  auto accel_flu = ftf::transform_frame_aircraft_baselink(accel_frd);
374 
375  publish_imu_data_raw(header, gyro_flu, accel_flu, accel_frd);
376  }
377  // [accel_available]
378 
382  // [mag_available]
383  if (imu_hr.fields_updated & (7 << 6)) {
384  auto mag_field = ftf::transform_frame_aircraft_baselink<Eigen::Vector3d>(
385  Eigen::Vector3d(imu_hr.xmag, imu_hr.ymag, imu_hr.zmag) * GAUSS_TO_TESLA);
386 
387  publish_mag(header, mag_field);
388  }
389  // [mag_available]
390 
394  // [static_pressure_available]
395  if (imu_hr.fields_updated & (1 << 9)) {
396  auto static_pressure_msg = boost::make_shared<sensor_msgs::FluidPressure>();
397 
398  static_pressure_msg->header = header;
399  static_pressure_msg->fluid_pressure = imu_hr.abs_pressure;
400 
401  static_press_pub.publish(static_pressure_msg);
402  }
403  // [static_pressure_available]
404 
408  // [differential_pressure_available]
409  if (imu_hr.fields_updated & (1 << 10)) {
410  auto differential_pressure_msg = boost::make_shared<sensor_msgs::FluidPressure>();
411 
412  differential_pressure_msg->header = header;
413  differential_pressure_msg->fluid_pressure = imu_hr.diff_pressure;
414 
415  diff_press_pub.publish(differential_pressure_msg);
416  }
417  // [differential_pressure_available]
418 
422  // [temperature_available]
423  if (imu_hr.fields_updated & (1 << 12)) {
424  auto temp_msg = boost::make_shared<sensor_msgs::Temperature>();
425 
426  temp_msg->header = header;
427  temp_msg->temperature = imu_hr.temperature;
428 
429  temp_imu_pub.publish(temp_msg);
430  }
431  // [temperature_available]
432  }

440  void handle_raw_imu(const mavlink::mavlink_message_t *msg, mavlink::common::msg::RAW_IMU &imu_raw)
441  {
442  ROS_INFO_COND_NAMED(!has_raw_imu, "imu", "IMU: Raw IMU message used.");
443  has_raw_imu = true;
444 
445  if (has_hr_imu || has_scaled_imu)
446  return;
447 
448  auto imu_msg = boost::make_shared<sensor_msgs::Imu>();
449  auto header = m_uas->synchronized_header(frame_id, imu_raw.time_usec);
450 
453  auto gyro_flu = ftf::transform_frame_aircraft_baselink<Eigen::Vector3d>(
454  Eigen::Vector3d(imu_raw.xgyro, imu_raw.ygyro, imu_raw.zgyro) * MILLIRS_TO_RADSEC);
455  auto accel_frd = Eigen::Vector3d(imu_raw.xacc, imu_raw.yacc, imu_raw.zacc);
456  auto accel_flu = ftf::transform_frame_aircraft_baselink<Eigen::Vector3d>(accel_frd);
457 
458  if (m_uas->is_ardupilotmega()) {
459  accel_frd *= MILLIG_TO_MS2;
460  accel_flu *= MILLIG_TO_MS2;
461  } else if (m_uas->is_px4()) {
462  accel_frd *= MILLIMS2_TO_MS2;
463  accel_flu *= MILLIMS2_TO_MS2;
464  }
465 
466  publish_imu_data_raw(header, gyro_flu, accel_flu, accel_frd);
467 
468  if (!m_uas->is_ardupilotmega()) {
469  ROS_WARN_THROTTLE_NAMED(60, "imu", "IMU: linear acceleration on RAW_IMU known on APM only.");
470  ROS_WARN_THROTTLE_NAMED(60, "imu", "IMU: ~imu/data_raw stores unscaled raw acceleration report.");
471  linear_accel_vec_flu.setZero();
472  linear_accel_vec_frd.setZero();
473  }
474 
478  // [mag_field]
479  auto mag_field = ftf::transform_frame_aircraft_baselink<Eigen::Vector3d>(
480  Eigen::Vector3d(imu_raw.xmag, imu_raw.ymag, imu_raw.zmag) * MILLIT_TO_TESLA);
481  // [mag_field]
482 
483  publish_mag(header, mag_field);
484  }
