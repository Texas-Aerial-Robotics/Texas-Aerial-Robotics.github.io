607  link_id = msgbuf[-13]

613  if stream_key in self.signing.stream_timestamps:
614  if timestamp <= self.signing.stream_timestamps[stream_key]:
615  # reject old timestamp
616  # print('old timestamp')
617  return False
618  else:
619  # a new stream has appeared. Accept the timestamp if it is at most
