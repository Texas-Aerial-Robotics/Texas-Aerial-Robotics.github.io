41 static void publish_track_marker(const geometry_msgs::PoseStamped::ConstPtr &pose)
42 {
43  static boost::shared_ptr<visualization_msgs::Marker> track_marker;
44 
45  if ( !track_marker )
46  {
47  track_marker = boost::make_shared<visualization_msgs::Marker>();
48  track_marker->type = visualization_msgs::Marker::CUBE_LIST;
49  track_marker->ns = "fcu";
50  track_marker->action = visualization_msgs::Marker::ADD;
51  track_marker->scale.x = marker_scale * 0.015;
52  track_marker->scale.y = marker_scale * 0.015;
53  track_marker->scale.z = marker_scale * 0.015;
54  track_marker->color.a = 1.0;
55  track_marker->color.r = 0.0;
56  track_marker->color.g = 0.0;
57  track_marker->color.b = 0.5;
58  track_marker->points.reserve(max_track_size);
59  }
60 
61  static int marker_idx = 0;
62 
63  if ( track_marker->points.size() < max_track_size )
64  track_marker->points.push_back(pose->pose.position);
65  else track_marker->points[marker_idx] = pose->pose.position;
66 
67  marker_idx = ++marker_idx % max_track_size;
68 
69  track_marker->header = pose->header;
70  track_marker_pub.publish(track_marker);

104 static void create_vehicle_markers( int num_rotors, float arm_len, float body_width, float body_height )
105 {
106  if ( num_rotors <= 0 ) num_rotors = 2;
107 
112  if ( vehicle_marker )
113  return;
114 
115  vehicle_marker = boost::make_shared<visualization_msgs::MarkerArray>();
116  vehicle_marker->markers.reserve( 2 * num_rotors + 1 );
117 
122  // rotor marker template
123  visualization_msgs::Marker rotor;
124  rotor.header.stamp = ros::Time();
125  rotor.header.frame_id = child_frame_id;
126  rotor.ns = "vehicle_rotor";
127  rotor.action = visualization_msgs::Marker::ADD;
128  rotor.type = visualization_msgs::Marker::CYLINDER;
129  rotor.scale.x = 0.2 * marker_scale;
130  rotor.scale.y = 0.2 * marker_scale;
131  rotor.scale.z = 0.01 * marker_scale;
132  rotor.color.r = 0.4;
133  rotor.color.g = 0.4;
134  rotor.color.b = 0.4;
135  rotor.color.a = 0.8;
136  rotor.pose.position.z = 0;
137 
138  // arm marker template
139  visualization_msgs::Marker arm;
140  arm.header.stamp = ros::Time();
141  arm.header.frame_id = child_frame_id;
142  arm.ns = "vehicle_arm";
143  arm.action = visualization_msgs::Marker::ADD;
144  arm.type = visualization_msgs::Marker::CUBE;
145  arm.scale.x = arm_len * marker_scale;
146  arm.scale.y = 0.02 * marker_scale;
147  arm.scale.z = 0.01 * marker_scale;
148  arm.color.r = 0.0;
149  arm.color.g = 0.0;
150  arm.color.b = 1.0;
151  arm.color.a = 1.0;
152  arm.pose.position.z = -0.015 * marker_scale;
153 
154  float angle_increment = 2 * M_PI / num_rotors;
155 
156  for ( float angle = angle_increment / 2; angle <= (2 * M_PI); angle += angle_increment )
157  {
158  rotor.pose.position.x = arm_len * cos(angle) * marker_scale;
159  rotor.pose.position.y = arm_len * sin(angle) * marker_scale;
160  rotor.id++;
161 
162  arm.pose.position.x = rotor.pose.position.x / 2;
163  arm.pose.position.y = rotor.pose.position.y / 2;
164  arm.pose.orientation = tf::createQuaternionMsgFromYaw(angle);
165  arm.id++;
166 
167  vehicle_marker->markers.push_back(rotor);
168  vehicle_marker->markers.push_back(arm);
169  }
170 
171  // body marker template
172  visualization_msgs::Marker body;
173  body.header.stamp = ros::Time();
174  body.header.frame_id = child_frame_id;
175  body.ns = "vehicle_body";
176  body.action = visualization_msgs::Marker::ADD;
177  body.type = visualization_msgs::Marker::CUBE;
178  body.scale.x = body_width * marker_scale;
179  body.scale.y = body_width * marker_scale;
180  body.scale.z = body_height * marker_scale;
181  body.color.r = 0.0;
182  body.color.g = 1.0;
183  body.color.b = 0.0;
184  body.color.a = 0.8;
185 
186  vehicle_marker->markers.push_back(body);
187 }
