174  void send_fake_gps(const ros::Time &stamp, const Eigen::Vector3d &ecef_offset) {
175  // Throttle incoming messages to 5hz
176  if ((ros::Time::now() - last_pos_time) < ros::Duration(gps_rate)) {
177  return;
178  }
179  last_pos_time = ros::Time::now();
180 
186  mavlink::common::msg::HIL_GPS fix {};
187 
188  Eigen::Vector3d geodetic;
189  Eigen::Vector3d current_ecef(ecef_origin.x() + ecef_offset.x(),
190  ecef_origin.y() + ecef_offset.y(),
191  ecef_origin.z() + ecef_offset.z());
192 
193  try {
194  earth.Reverse(current_ecef.x(), current_ecef.y(), current_ecef.z(),
195  geodetic.x(), geodetic.y(), geodetic.z());
196  }
197  catch (const std::exception& e) {
198  ROS_INFO_STREAM("FGPS: Caught exception: " << e.what() << std::endl);
199  }
200 
201  Eigen::Vector3d vel = ((old_ecef - current_ecef) / (stamp.toSec() - old_stamp)) * 1e2;// [cm/s]
202 
203  // compute course over ground
204  double cog;
205  if (vel.x() == 0 && vel.y() == 0) {
206  cog = 0;
207  }
208  else if (vel.x() >= 0 && vel.y() < 0) {
209  cog = M_PI * 5 / 2 - atan2(vel.x(), vel.y());
210  }
211  else {
212  cog = M_PI / 2 - atan2(vel.x(), vel.y());
213  }
214 
215  // Fill in and send message
216  fix.time_usec = stamp.toNSec() / 1000; // [useconds]
217  fix.lat = geodetic.x() * 1e7; // [degrees * 1e7]
218  fix.lon = geodetic.y() * 1e7; // [degrees * 1e7]
219  fix.alt = (geodetic.z() + GeographicLib::Geoid::ELLIPSOIDTOGEOID *
220  (*m_uas->egm96_5)(geodetic.x(), geodetic.y())) * 1e3; // [meters * 1e3]
221  fix.vel = vel.block<2, 1>(0, 0).norm(); // [cm/s]
222  fix.vn = vel.x(); // [cm/s]
223  fix.ve = vel.y(); // [cm/s]
224  fix.vd = vel.z(); // [cm/s]
225  fix.cog = cog * 1e2; // [degrees * 1e2]
226  fix.eph = eph * 1e2; // [cm]
227  fix.epv = epv * 1e2; // [cm]
228  fix.fix_type = utils::enum_value(fix_type);;
229  fix.satellites_visible = satellites_visible;
230 
231  // store old values
232  old_stamp = stamp.toSec();
233  old_ecef = current_ecef;
234 
235  UAS_FCU(m_uas)->send_message_ignore_drop(fix);
236  }
