200  void trajectory_cb(const mavros_msgs::Trajectory::ConstPtr &req)
201  {
202  ROS_ASSERT(NUM_POINTS == req->point_valid.size());
203 
204  mavlink::common::msg::TRAJECTORY_REPRESENTATION_WAYPOINTS trajectory {};
205 
206  auto fill_point = [&](mavlink::common::msg::TRAJECTORY_REPRESENTATION_WAYPOINTS & t, const RosPoints &rp, const size_t i) {
207  const auto valid = req->point_valid[i];
208 
209  auto valid_so_far = trajectory.valid_points;
210  if (!valid) {
211  fill_points_all_unused(t, i);
212  return;
213  }
214 
215  trajectory.valid_points = valid_so_far + 1;
216  fill_points_position(t.pos_x, t.pos_y, t.pos_z, rp.position, i);
217  fill_points_velocity(t.vel_x, t.vel_y, t.vel_z, rp.velocity, i);
218  fill_points_acceleration(t.acc_x, t.acc_y, t.acc_z, rp.acceleration_or_force, i);
219  fill_points_yaw_wp(t.pos_yaw, rp.yaw, i);
220  fill_points_yaw_speed(t.vel_yaw, rp.yaw_rate, i);
221  };
222 
223  trajectory.time_usec = req->header.stamp.toNSec() / 1000;

284  void handle_trajectory(const mavlink::mavlink_message_t *msg, mavlink::common::msg::TRAJECTORY_REPRESENTATION_WAYPOINTS &trajectory)
285  {
286  auto tr_desired = boost::make_shared<mavros_msgs::Trajectory>();
287 
288  auto fill_msg_point = [&](RosPoints & p, const mavlink::common::msg::TRAJECTORY_REPRESENTATION_WAYPOINTS & t, const size_t i) {
289  fill_msg_position(p.position, t, i);
290  fill_msg_velocity(p.velocity, t, i);
291  fill_msg_acceleration(p.acceleration_or_force, t, i);
292  p.yaw = wrap_pi((M_PI / 2.0f) - t.pos_yaw[i]);
293  p.yaw_rate = t.vel_yaw[i];
294  };
295 
296  tr_desired->header = m_uas->synchronized_header("local_origin", trajectory.time_usec);
297 
298  for (int i = 0; i < trajectory.valid_points; ++i)
299  {
300  tr_desired->point_valid[i] = true;
301  }
302 
303  for (int i = trajectory.valid_points; i < NUM_POINTS; ++i)
304  {
305  tr_desired->point_valid[i] = false;
306  }
307 
308  // [[[cog:
309  // for i in range(5):
310  // cog.outl(

312  // .format(i0=i, i1=i + 1, )
313  // )
314  // ]]]

325  float wrap_pi(float a)
326  {
327  if (!std::isfinite(a)) {
