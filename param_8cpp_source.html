417  void handle_param_value(const mavlink::mavlink_message_t *msg, mavlink::common::msg::PARAM_VALUE &pmsg)
418  {
419  lock_guard lock(mutex);
420 
421  auto param_id = mavlink::to_string(pmsg.param_id);
422 
423  // search
424  auto param_it = parameters.find(param_id);
425  if (param_it != parameters.end()) {
426  // parameter exists
427  auto &p = param_it->second;
428 
429  if (m_uas->is_ardupilotmega())
430  p.set_value_apm_quirk(pmsg);
431  else
432  p.set_value(pmsg);
433 
434  // check that ack required
435  auto set_it = set_parameters.find(param_id);
436  if (set_it != set_parameters.end()) {
437  set_it->second->ack.notify_all();
438  }
439 
440  ROS_WARN_STREAM_COND_NAMED(
441  ((p.param_index != pmsg.param_index &&
442  pmsg.param_index != UINT16_MAX) ||
443  p.param_count != pmsg.param_count),
444  "param",
445  "PR: Param " << p.to_string() << " different index: " << pmsg.param_index << "/" << pmsg.param_count);
446  ROS_DEBUG_STREAM_NAMED("param", "PR: Update param " << p.to_string());
447  }
448  else {
449  // insert new element
450  Parameter p{};
451  p.param_id = param_id;
452  p.param_index = pmsg.param_index;
453  p.param_count = pmsg.param_count;
454 
455  if (m_uas->is_ardupilotmega())
456  p.set_value_apm_quirk(pmsg);
457  else
458  p.set_value(pmsg);
459 
460  parameters[param_id] = p;
461 
462  ROS_DEBUG_STREAM_NAMED("param", "PR: New param " << p.to_string());
463  }
464 
465  if (param_state == PR::RXLIST || param_state == PR::RXPARAM) {
466  // we received first param. setup list timeout
467  if (param_state == PR::RXLIST) {
468  param_count = pmsg.param_count;
469  param_state = PR::RXPARAM;
470 
471  parameters_missing_idx.clear();
472  if (param_count != UINT16_MAX) {
473  ROS_DEBUG_NAMED("param", "PR: waiting %zu parameters", param_count);
474  // declare that all parameters are missing
475  for (uint16_t idx = 0; idx < param_count; idx++)
476  parameters_missing_idx.push_back(idx);
477  }
478  else
479  ROS_WARN_NAMED("param", "PR: FCU does not know index for first element! "
480  "Param list may be truncated.");
481  }
482 
483  // trying to avoid endless rerequest loop
484  // Issue #276
485  bool it_is_first_requested = parameters_missing_idx.front() == pmsg.param_index;
486 
487  // remove idx for that message
488  parameters_missing_idx.remove(pmsg.param_index);
489 
490  // in receiving mode we use param_rx_retries for LIST and PARAM
491  if (it_is_first_requested)
492  param_rx_retries = RETRIES_COUNT;
493 
494  restart_timeout_timer();
495 
496  /* index starting from 0, receivig done */
497  if (parameters_missing_idx.empty()) {
498  ssize_t missed = param_count - parameters.size();
499  ROS_INFO_COND_NAMED(missed == 0, "param", "PR: parameters list received");
500  ROS_WARN_COND_NAMED(missed > 0, "param",
501  "PR: parameters list received, but %zd parametars are missed",
502  missed);
503  go_idle();
504  list_receiving.notify_all();
505  }
506  }

593  void timeout_cb(const ros::TimerEvent &event)
594  {
595  lock_guard lock(mutex);
596  if (param_state == PR::RXLIST && param_rx_retries > 0) {
597  param_rx_retries--;
598  ROS_WARN_NAMED("param", "PR: request list timeout, retries left %zu", param_rx_retries);
599 
600  restart_timeout_timer();
601  param_request_list();
602  }
603  else if (param_state == PR::RXPARAM) {
604  if (parameters_missing_idx.empty()) {
605  ROS_WARN_NAMED("param", "PR: missing list is clear, but we in RXPARAM state, "
606  "maybe last rerequest fails. Params missed: %zd",
607  param_count - parameters.size());
608  go_idle();
609  list_receiving.notify_all();
610  return;
611  }
612 
613  uint16_t first_miss_idx = parameters_missing_idx.front();
614  if (param_rx_retries > 0) {
615  param_rx_retries--;
616  ROS_WARN_NAMED("param", "PR: request param #%u timeout, retries left %zu, and %zu params still missing",
617  first_miss_idx, param_rx_retries, parameters_missing_idx.size());
618  restart_timeout_timer();
619  param_request_read("", first_miss_idx);
620  }
621  else {
622  ROS_ERROR_NAMED("param", "PR: request param #%u completely missing.", first_miss_idx);
623  parameters_missing_idx.pop_front();
624  restart_timeout_timer();
625  if (!parameters_missing_idx.empty()) {
626  param_rx_retries = RETRIES_COUNT;
627  first_miss_idx = parameters_missing_idx.front();
628 
629  ROS_WARN_NAMED("param", "PR: %zu params still missing, trying to request next: #%u",
630  parameters_missing_idx.size(), first_miss_idx);
631  param_request_read("", first_miss_idx);
632  }
633  }
634  }
635  else if (param_state == PR::TXPARAM) {
636  auto it = set_parameters.begin();
637  if (it == set_parameters.end()) {
638  ROS_DEBUG_NAMED("param", "PR: send list empty, but state TXPARAM");
639  go_idle();
640  return;
641  }
642 
643  if (it->second->retries_remaining > 0) {
644  it->second->retries_remaining--;
645  ROS_WARN_NAMED("param", "PR: Resend param set for %s, retries left %zu",
646  it->second->param.param_id.c_str(),
647  it->second->retries_remaining);
648  restart_timeout_timer();
649  param_set(it->second->param);
650  }
651  else {
652  ROS_ERROR_NAMED("param", "PR: Param set for %s timed out.",
653  it->second->param.param_id.c_str());
654  it->second->is_timedout = true;
655  it->second->ack.notify_all();
656  }
657  }
658  else {
659  ROS_DEBUG_NAMED("param", "PR: timeout in IDLE!");
660  }
661  }
