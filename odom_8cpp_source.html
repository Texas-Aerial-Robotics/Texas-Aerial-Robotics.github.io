114  void odom_cb(const nav_msgs::Odometry::ConstPtr &odom)
115  {
120  Eigen::Affine3d tf_parent2local;
121  Eigen::Affine3d tf_child2local;
122  Eigen::Affine3d tf_parent2body;
123  Eigen::Affine3d tf_child2body;
124 
125  try {
126  // transform lookup WRT local frame
127  tf_parent2local = tf2::transformToEigen(m_uas->tf2_buffer.lookupTransform(
128  odom->header.frame_id, local_frame_orientation,
129  ros::Time(0)));
130  tf_child2local = tf2::transformToEigen(m_uas->tf2_buffer.lookupTransform(
131  odom->child_frame_id, local_frame_orientation,
132  ros::Time(0)));
133 
134  // transform lookup WRT body frame
135  tf_parent2body = tf2::transformToEigen(m_uas->tf2_buffer.lookupTransform(
136  odom->header.frame_id, body_frame_orientation,
137  ros::Time(0)));
138  tf_child2body = tf2::transformToEigen(m_uas->tf2_buffer.lookupTransform(
139  odom->child_frame_id, body_frame_orientation,
140  ros::Time(0)));
141  } catch (tf2::TransformException &ex) {
142  ROS_ERROR_THROTTLE_NAMED(1, "odom", "ODOM: Ex: %s", ex.what());
143  return;
144  }
145 
147  ftf::Covariance6d cov_pose {}; // zero initialized
148  ftf::EigenMapCovariance6d cov_pose_map(cov_pose.data());
149 
151  ftf::Covariance6d cov_vel {}; // zero initialized
152  ftf::EigenMapCovariance6d cov_vel_map(cov_vel.data());
153 
157  Eigen::Vector3d position {};
158  Eigen::Quaterniond orientation {};
159  Eigen::Vector3d lin_vel {};
160  Eigen::Vector3d ang_vel {};
161  Matrix6d r_pose = Matrix6d::Zero();
162  Matrix6d r_vel = Matrix6d::Zero();
163 
164  mavlink::common::msg::ODOMETRY msg {};
165 
172  position = Eigen::Vector3d(tf_parent2local.linear() * ftf::to_eigen(odom->pose.pose.position));
173 
174  // Orientation represented by a quaternion rotation from the local frame to XYZ body frame
175  Eigen::Quaterniond q_parent2child(ftf::to_eigen(odom->pose.pose.orientation));
176  Eigen::Affine3d tf_local2body = tf_parent2local * q_parent2child * tf_child2body.inverse();
177  orientation = Eigen::Quaterniond(tf_local2body.linear());
178 
179  r_pose.block<3, 3>(0, 0) = r_pose.block<3, 3>(3, 3) = tf_parent2local.linear();
180 
181  msg.frame_id = utils::enum_value(lf_id);
182 
187  auto set_tf = [&](Eigen::Affine3d affineTf, MAV_FRAME frame_id) {
188  lin_vel = Eigen::Vector3d(affineTf.linear() * ftf::to_eigen(odom->twist.twist.linear));
189  ang_vel = Eigen::Vector3d(affineTf.linear() * ftf::to_eigen(odom->twist.twist.angular));
190  r_vel.block<3, 3>(0, 0) = r_vel.block<3, 3>(3, 3) = affineTf.linear();
191 
192  msg.child_frame_id = utils::enum_value(frame_id);
193  };
194 
195  if (odom->child_frame_id == "world" || odom->child_frame_id == "odom") {
196  // the child_frame_id would be the same reference frame as frame_id
197  set_tf(tf_child2local, lf_id);
198  }
199  else {
200  // the child_frame_id would be the WRT a body frame reference
201  set_tf(tf_child2body, bf_id);
202  }
203 
205  cov_pose_map = r_pose * cov_pose_map * r_pose.transpose();
206  cov_vel_map = r_vel * cov_vel_map * r_vel.transpose();
207 
208  ROS_DEBUG_STREAM_NAMED("odom", "ODOM: pose covariance matrix:" << std::endl << cov_pose_map);
209  ROS_DEBUG_STREAM_NAMED("odom", "ODOM: velocity covariance matrix:" << std::endl << cov_vel_map);
210 
211  /* -*- ODOMETRY msg parser -*- */
212  msg.time_usec = odom->header.stamp.toNSec() / 1e3;
213 
214  // [[[cog:
215  // for a, b in (('', 'position'), ('v', 'lin_vel')):
216  // for f in 'xyz':
217  // cog.outl("msg.{a}{f} = {b}.{f}();".format(**locals()))
218  // for a, b in zip("xyz", ('rollspeed', 'pitchspeed', 'yawspeed')):
219  // cog.outl("msg.{b} = ang_vel.{a}();".format(**locals()))
220  // ]]]
221  msg.x = position.x();
222  msg.y = position.y();
223  msg.z = position.z();
224  msg.vx = lin_vel.x();
225  msg.vy = lin_vel.y();
226  msg.vz = lin_vel.z();
227  msg.rollspeed = ang_vel.x();
228  msg.pitchspeed = ang_vel.y();
229  msg.yawspeed = ang_vel.z();
230  // [[[end]]] (checksum: ead24a1a6a14496c9de6c1951ccfbbd7)

232  ftf::quaternion_to_mavlink(orientation, msg.q);
233  ftf::covariance_urt_to_mavlink(cov_pose_map, msg.pose_covariance);
234  ftf::covariance_urt_to_mavlink(cov_vel_map, msg.twist_covariance);
