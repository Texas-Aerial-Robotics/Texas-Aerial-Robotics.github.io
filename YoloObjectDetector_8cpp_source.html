328 void *YoloObjectDetector::detectInThread()
329 {
330  running_ = 1;
331  float nms = .4;
332 
333  layer l = net_->layers[net_->n - 1];
334  float *X = buffLetter_[(buffIndex_ + 2) % 3].data;
335  float *prediction = network_predict(net_, X);
336 
337  rememberNetwork(net_);
338  detection *dets = 0;
339  int nboxes = 0;
340  dets = avgPredictions(net_, &nboxes);
341 
342  if (nms > 0) do_nms_obj(dets, nboxes, l.classes, nms);
343 
344  if (enableConsoleOutput_) {
345  printf("\033[2J");
346  printf("\033[1;1H");
347  printf("\nFPS:%.1f\n",fps_);
348  printf("Objects:\n\n");
349  }
350  image display = buff_[(buffIndex_+2) % 3];
351  draw_detections(display, dets, nboxes, demoThresh_, demoNames_, demoAlphabet_, demoClasses_);
352 
353  // extract the bounding boxes and send them to ROS
354  int i, j;
355  int count = 0;
356  for (i = 0; i < nboxes; ++i) {
357  float xmin = dets[i].bbox.x - dets[i].bbox.w / 2.;
358  float xmax = dets[i].bbox.x + dets[i].bbox.w / 2.;
359  float ymin = dets[i].bbox.y - dets[i].bbox.h / 2.;
360  float ymax = dets[i].bbox.y + dets[i].bbox.h / 2.;
361 
362  if (xmin < 0)
363  xmin = 0;
364  if (ymin < 0)
365  ymin = 0;
366  if (xmax > 1)
367  xmax = 1;
368  if (ymax > 1)
369  ymax = 1;
370 
371  // iterate through possible boxes and collect the bounding boxes
372  for (j = 0; j < demoClasses_; ++j) {
373  if (dets[i].prob[j]) {
374  float x_center = (xmin + xmax) / 2;
375  float y_center = (ymin + ymax) / 2;
376  float BoundingBox_width = xmax - xmin;
377  float BoundingBox_height = ymax - ymin;
378 
379  // define bounding box
380  // BoundingBox must be 1% size of frame (3.2x2.4 pixels)
381  if (BoundingBox_width > 0.01 && BoundingBox_height > 0.01) {
382  roiBoxes_[count].x = x_center;
383  roiBoxes_[count].y = y_center;
384  roiBoxes_[count].w = BoundingBox_width;
385  roiBoxes_[count].h = BoundingBox_height;
386  roiBoxes_[count].Class = j;
387  roiBoxes_[count].prob = dets[i].prob[j];
388  count++;
389  }
390  }
391  }
392  }
393 
394  // create array to store found bounding boxes
395  // if no object detected, make sure that ROS knows that num = 0
396  if (count == 0) {
397  roiBoxes_[0].num = 0;
398  } else {
399  roiBoxes_[0].num = count;
400  }
401 
402  free_detections(dets, nboxes);
403  demoIndex_ = (demoIndex_ + 1) % demoFrame_;
404  running_ = 0;
405  return 0;
406 }

480 {
481  const auto wait_duration = std::chrono::milliseconds(2000);
482  while (!getImageStatus()) {
483  printf("Waiting for image.\n");
484  if (!isNodeRunning()) {
485  return;
486  }
487  std::this_thread::sleep_for(wait_duration);
488  }
489 
490  std::thread detect_thread;
491  std::thread fetch_thread;
492 
493  srand(2222222);
494 
495  int i;
496  demoTotal_ = sizeNetwork(net_);
497  predictions_ = (float **) calloc(demoFrame_, sizeof(float*));
498  for (i = 0; i < demoFrame_; ++i){
499  predictions_[i] = (float *) calloc(demoTotal_, sizeof(float));
500  }
501  avg_ = (float *) calloc(demoTotal_, sizeof(float));
502 
503  layer l = net_->layers[net_->n - 1];
504  roiBoxes_ = (darknet_ros::RosBox_ *) calloc(l.w * l.h * l.n, sizeof(darknet_ros::RosBox_));
505 
506  IplImageWithHeader_ imageAndHeader = getIplImageWithHeader();
507  IplImage* ROS_img = imageAndHeader.image;
508  buff_[0] = ipl_to_image(ROS_img);
509  buff_[1] = copy_image(buff_[0]);
510  buff_[2] = copy_image(buff_[0]);
511  headerBuff_[0] = imageAndHeader.header;
512  headerBuff_[1] = headerBuff_[0];
513  headerBuff_[2] = headerBuff_[0];
514  buffLetter_[0] = letterbox_image(buff_[0], net_->w, net_->h);
515  buffLetter_[1] = letterbox_image(buff_[0], net_->w, net_->h);
516  buffLetter_[2] = letterbox_image(buff_[0], net_->w, net_->h);
517  ipl_ = cvCreateImage(cvSize(buff_[0].w, buff_[0].h), IPL_DEPTH_8U, buff_[0].c);
518 
519  int count = 0;
520 
521  if (!demoPrefix_ && viewImage_) {
522  cvNamedWindow("YOLO V3", CV_WINDOW_NORMAL);
523  if (fullScreen_) {
524  cvSetWindowProperty("YOLO V3", CV_WND_PROP_FULLSCREEN, CV_WINDOW_FULLSCREEN);
525  } else {
526  cvMoveWindow("YOLO V3", 0, 0);
527  cvResizeWindow("YOLO V3", 640, 480);
528  }
529  }
530 
531  demoTime_ = what_time_is_it_now();
532 
533  while (!demoDone_) {
534  buffIndex_ = (buffIndex_ + 1) % 3;
535  fetch_thread = std::thread(&YoloObjectDetector::fetchInThread, this);
536  detect_thread = std::thread(&YoloObjectDetector::detectInThread, this);
537  if (!demoPrefix_) {
538  fps_ = 1./(what_time_is_it_now() - demoTime_);
539  demoTime_ = what_time_is_it_now();
540  if (viewImage_) {
541  displayInThread(0);
542  }
543  publishInThread();
544  } else {
545  char name[256];
546  sprintf(name, "%s_%08d", demoPrefix_, count);
547  save_image(buff_[(buffIndex_ + 1) % 3], name);
548  }
549  fetch_thread.join();
550  detect_thread.join();
551  ++count;
552  if (!isNodeRunning()) {
553  demoDone_ = true;
554  }
555  }
556 
557 }

579 void *YoloObjectDetector::publishInThread()
580 {
581  // Publish image.
582  cv::Mat cvImage = cv::cvarrToMat(ipl_);
583  if (!publishDetectionImage(cv::Mat(cvImage))) {
584  ROS_DEBUG("Detection image has not been broadcasted.");
585  }
586 
587  // Publish bounding boxes and detection result.
588  int num = roiBoxes_[0].num;
589  if (num > 0 && num <= 100) {
590  for (int i = 0; i < num; i++) {
591  for (int j = 0; j < numClasses_; j++) {
592  if (roiBoxes_[i].Class == j) {
593  rosBoxes_[j].push_back(roiBoxes_[i]);
594  rosBoxCounter_[j]++;
595  }
596  }
597  }
598 
599  std_msgs::Int8 msg;
600  msg.data = num;
601  objectPublisher_.publish(msg);
602 
603  for (int i = 0; i < numClasses_; i++) {
604  if (rosBoxCounter_[i] > 0) {
605  darknet_ros_msgs::BoundingBox boundingBox;
606 
607  for (int j = 0; j < rosBoxCounter_[i]; j++) {
608  int xmin = (rosBoxes_[i][j].x - rosBoxes_[i][j].w / 2) * frameWidth_;
609  int ymin = (rosBoxes_[i][j].y - rosBoxes_[i][j].h / 2) * frameHeight_;
610  int xmax = (rosBoxes_[i][j].x + rosBoxes_[i][j].w / 2) * frameWidth_;
611  int ymax = (rosBoxes_[i][j].y + rosBoxes_[i][j].h / 2) * frameHeight_;
612 
613  boundingBox.Class = classLabels_[i];
614  boundingBox.probability = rosBoxes_[i][j].prob;
615  boundingBox.xmin = xmin;
616  boundingBox.ymin = ymin;
617  boundingBox.xmax = xmax;
618  boundingBox.ymax = ymax;
619  boundingBoxesResults_.bounding_boxes.push_back(boundingBox);
620  }
621  }
622  }
623  boundingBoxesResults_.header.stamp = ros::Time::now();
624  boundingBoxesResults_.header.frame_id = "detection";
625  boundingBoxesResults_.image_header = headerBuff_[(buffIndex_ + 1) % 3];
626  boundingBoxesPublisher_.publish(boundingBoxesResults_);
627  } else {
628  std_msgs::Int8 msg;
629  msg.data = 0;
630  objectPublisher_.publish(msg);
631  }
632  if (isCheckingForObjects()) {
633  ROS_DEBUG("[YoloObjectDetector] check for objects in image.");
634  darknet_ros_msgs::CheckForObjectsResult objectsActionResult;
