122  void send_position_target(const ros::Time &stamp, const Eigen::Affine3d &tr)
123  {
124  using mavlink::common::MAV_FRAME;
125 
126  /* Documentation start from bit 1 instead 0;
127  * Ignore velocity and accel vectors, yaw rate.
128  *
129  * In past versions on PX4 there been bug described in #273.
130  * If you got similar issue please try update firmware first.
131  */
132  const uint16_t ignore_all_except_xyz_y = (1 << 11) | (7 << 6) | (7 << 3);
133 
134  auto p = [&] () {
135  if (static_cast<MAV_FRAME>(mav_frame) == MAV_FRAME::BODY_NED || static_cast<MAV_FRAME>(mav_frame) == MAV_FRAME::BODY_OFFSET_NED) {
136  return ftf::transform_frame_baselink_aircraft(Eigen::Vector3d(tr.translation()));
137  } else {
138  return ftf::transform_frame_enu_ned(Eigen::Vector3d(tr.translation()));
139  }
140  } ();
141 
142  auto q = [&] () {
143  if (mav_frame == MAV_FRAME::BODY_NED || mav_frame == MAV_FRAME::BODY_OFFSET_NED) {
144  return ftf::transform_orientation_baselink_aircraft(Eigen::Quaterniond(tr.rotation()));
145  } else {
146  return ftf::transform_orientation_enu_ned(
147  ftf::transform_orientation_baselink_aircraft(Eigen::Quaterniond(tr.rotation())));
148  }
149  } ();
150 
151  set_position_target_local_ned(stamp.toNSec() / 1000000,
152  utils::enum_value(mav_frame),
153  ignore_all_except_xyz_y,
154  p,
155  Eigen::Vector3d::Zero(),
156  Eigen::Vector3d::Zero(),
157  ftf::quaternion_get_yaw(q), 0.0);
158  }

184  void setpointg_cb(const mavros_msgs::GlobalPositionTarget::ConstPtr &req)
185  {
194  GeographicLib::Geocentric earth(GeographicLib::Constants::WGS84_a(), GeographicLib::Constants::WGS84_f());
195 
196  Eigen::Vector3d goal_gps(req->latitude, req->longitude, req->altitude);
197 
198  // current gps -> curent ECEF
199  Eigen::Vector3d current_ecef;
200  earth.Forward(current_gps.x(), current_gps.y(), current_gps.z(),
201  current_ecef.x(), current_ecef.y(), current_ecef.z());
202 
203  // goal gps -> goal ECEF
204  Eigen::Vector3d goal_ecef;
205  earth.Forward(goal_gps.x(), goal_gps.y(), goal_gps.z(),
206  goal_ecef.x(), goal_ecef.y(), goal_ecef.z());
207 
208  // get ENU offset from ECEF offset
209  Eigen::Vector3d ecef_offset = goal_ecef - current_ecef;
210  Eigen::Vector3d enu_offset = ftf::transform_frame_ecef_enu(ecef_offset, current_gps);
211 
212  // prepare yaw angle
213  Eigen::Affine3d sp; // holds position setpoint
214  Eigen::Quaterniond q; // holds desired yaw
215 
216  q = Eigen::AngleAxisd(0.0, Eigen::Vector3d::UnitX()) // roll
217  * Eigen::AngleAxisd(0.0, Eigen::Vector3d::UnitY()) // pitch
218  * Eigen::AngleAxisd(req->yaw, Eigen::Vector3d::UnitZ()); // yaw
219 
220  // set position setpoint
221  sp.translation() = current_local_pos + enu_offset;
222  // set desired orientation
223  sp.linear() = q.toRotationMatrix();
224 
225  // Only send if current gps is updated, to avoid divergence
226  if ((req->header.stamp.toNSec() / 1000000) > old_gps_stamp) {
227  old_gps_stamp = req->header.stamp.toNSec() / 1000000;
228  send_position_target(req->header.stamp, sp);
229  }
230  else {
231  ROS_WARN_THROTTLE_NAMED(10, "spgp", "SPG: sp not sent.");
232  }
233  }
