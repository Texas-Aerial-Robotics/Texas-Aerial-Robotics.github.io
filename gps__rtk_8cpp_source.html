55  void rtcm_cb(const mavros_msgs::RTCM::ConstPtr &msg)
56  {
57  mavlink::common::msg::GPS_RTCM_DATA rtcm_data;
58  const size_t max_frag_len = rtcm_data.data.size();
59 
60  uint8_t seq_u5 = uint8_t(msg->header.seq & 0x1F) << 3;
61 
62  if (msg->data.size() > 4 * max_frag_len) {
63  ROS_FATAL("gps_rtk: RTCM message received is bigger than the maximal possible size.");
64  return;
65  }
66 
67  auto data_it = msg->data.begin();
68  auto end_it = msg->data.end();
69 
70  if (msg->data.size() <= max_frag_len) {
71  rtcm_data.len = msg->data.size();
72  rtcm_data.flags = seq_u5;
73  std::copy(data_it, end_it, rtcm_data.data.begin());
74  std::fill(rtcm_data.data.begin() + rtcm_data.len, rtcm_data.data.end(), 0);
75  UAS_FCU(m_uas)->send_message(rtcm_data);
76  } else {
77  for (uint8_t fragment_id = 0; fragment_id < 4 && data_it < end_it; fragment_id++) {
78  uint8_t len = std::min((size_t) std::distance(data_it, end_it), max_frag_len);
79  rtcm_data.flags = 1; // LSB set indicates message is fragmented
80  rtcm_data.flags |= fragment_id++ << 1; // Next 2 bits are fragment id
81  rtcm_data.flags |= seq_u5; // Next 5 bits are sequence id
82  rtcm_data.len = len;
83 
84  std::copy(data_it, data_it + len, rtcm_data.data.begin());
85  std::fill(rtcm_data.data.begin() + len, rtcm_data.data.end(), 0);
86  UAS_FCU(m_uas)->send_message(rtcm_data);
87  std::advance(data_it, len);
88  }
89  }
90  }
91 };
