387 def message_checksum(msg):
388  '''calculate a 8-bit checksum of the key fields of a message, so we
389  can detect incompatible XML changes'''
390  from .mavcrc import x25crc
391  crc = x25crc()
392  crc.accumulate_str(msg.name + ' ')
393  # in order to allow for extensions the crc does not include
394  # any field extensions
395  crc_end = msg.base_fields()
396  for i in range(crc_end):
397  f = msg.ordered_fields[i]
398  crc.accumulate_str(f.type + ' ')
399  crc.accumulate_str(f.name + ' ')
400  if f.array_length:
401  crc.accumulate([f.array_length])
402  return (crc.crc&0xFF) ^ (crc.crc>>8)
