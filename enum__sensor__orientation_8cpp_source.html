140 int sensor_orientation_from_str(const std::string &sensor_orientation)
141 {
142  // XXX bsearch
143 
144  // 1. try to find by name
145  for (size_t idx = 0; idx < sensor_orientations.size(); idx++) {
146  if (sensor_orientations[idx].first == sensor_orientation)
147  return idx;
148  }
149 
150  // 2. try convert integer
151  // fallback for old configs that uses numeric orientation.
152  try {
153  int idx = std::stoi(sensor_orientation, 0, 0);
154  if (0 > idx || size_t(idx) > sensor_orientations.size()) {
155  ROS_ERROR_NAMED("uas", "SENSOR: orientation index out of bound: %d", idx);
156  return -1;
157  }
158  else
159  return idx;
160  }
161  catch (std::invalid_argument &ex) {
162  // failed
163  }
164 
165  ROS_ERROR_STREAM_NAMED("uas", "SENSOR: wrong orientation str: " << sensor_orientation);
166 
167  return -1;
168 }
