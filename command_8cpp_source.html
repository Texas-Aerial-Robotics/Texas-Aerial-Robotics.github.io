104  void handle_command_ack(const mavlink::mavlink_message_t *msg, mavlink::common::msg::COMMAND_ACK &ack)
105  {
106  lock_guard lock(mutex);
107  for (auto it = ack_waiting_list.cbegin();
108  it != ack_waiting_list.cend(); it++)
109  if ((*it)->expected_command == ack.command) {
110  (*it)->result = ack.result;
111  (*it)->ack.notify_all();
112  return;
113  }
114 
115  ROS_WARN_THROTTLE_NAMED(10, "cmd", "CMD: Unexpected command %u, result %u",
116  ack.command, ack.result);
117  }

121  bool wait_ack_for(CommandTransaction *tr) {
122  unique_lock lock(tr->cond_mutex);
123 
124  return tr->ack.wait_for(lock, std::chrono::nanoseconds(ACK_TIMEOUT_DT.toNSec()))
125  == std::cv_status::no_timeout;
126  }

133  bool send_command_long_and_wait(bool broadcast,
134  uint16_t command, uint8_t confirmation,
135  float param1, float param2,
136  float param3, float param4,
137  float param5, float param6,
138  float param7,
139  unsigned char &success, uint8_t &result)
140  {
141  using mavlink::common::MAV_RESULT;
142 
143  unique_lock lock(mutex);
144 
145  /* check transactions */
146  for (auto it = ack_waiting_list.cbegin();
147  it != ack_waiting_list.cend(); it++)
148  if ((*it)->expected_command == command) {
149  ROS_WARN_THROTTLE_NAMED(10, "cmd", "CMD: Command %u alredy in progress", command);
150  return false;
151  }
152 
157  bool is_ack_required = confirmation != 0 || m_uas->is_ardupilotmega() || m_uas->is_px4();
158  if (is_ack_required && !broadcast)
159  ack_waiting_list.push_back(new CommandTransaction(command));
160 
161  command_long(broadcast,
162  command, confirmation,
163  param1, param2,
164  param3, param4,
165  param5, param6,
166  param7);
167 
168  if (is_ack_required && !broadcast) {
169  auto it = ack_waiting_list.begin();
170  for (; it != ack_waiting_list.end(); it++)
171  if ((*it)->expected_command == command)
172  break;
173 
174  if (it == ack_waiting_list.end()) {
175  ROS_ERROR_NAMED("cmd", "CMD: CommandTransaction not found for %u", command);
176  return false;
177  }
178 
179  lock.unlock();
180  bool is_not_timeout = wait_ack_for(*it);
181  lock.lock();
182 
183  success = is_not_timeout && (*it)->result == enum_value(MAV_RESULT::ACCEPTED);
184  result = (*it)->result;
185 
186  delete *it;
187  ack_waiting_list.erase(it);
188  }
189  else {
190  success = true;
191  result = enum_value(MAV_RESULT::ACCEPTED);
192  }
193 
194  return true;
195  }
