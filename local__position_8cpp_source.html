83  void handle_local_position_ned(const mavlink::mavlink_message_t *msg, mavlink::common::msg::LOCAL_POSITION_NED &pos_ned)
84  {
85  //--------------- Transform FCU position and Velocity Data ---------------//
86  auto enu_position = ftf::transform_frame_ned_enu(Eigen::Vector3d(pos_ned.x, pos_ned.y, pos_ned.z));
87  auto enu_velocity = ftf::transform_frame_ned_enu(Eigen::Vector3d(pos_ned.vx, pos_ned.vy, pos_ned.vz));
88 
89  //--------------- Get Odom Information ---------------//
90  // Note this orientation describes baselink->ENU transform
91  auto enu_orientation_msg = m_uas->get_attitude_orientation_enu();
92  auto baselink_angular_msg = m_uas->get_attitude_angular_velocity_enu();
93  Eigen::Quaterniond enu_orientation;
94  tf::quaternionMsgToEigen(enu_orientation_msg, enu_orientation);
95  auto baselink_linear = ftf::transform_frame_enu_baselink(enu_velocity, enu_orientation.inverse());
96 
97  //--------------- Generate Message Pointers ---------------//
98  auto pose = boost::make_shared<geometry_msgs::PoseStamped>();
99  auto twist = boost::make_shared<geometry_msgs::TwistStamped>();
100  auto odom = boost::make_shared<nav_msgs::Odometry>();
101 
102  pose->header = m_uas->synchronized_header(frame_id, pos_ned.time_boot_ms);
103  twist->header = pose->header;
104 
105  tf::pointEigenToMsg(enu_position, pose->pose.position);
106  pose->pose.orientation = enu_orientation_msg;
107 
108  tf::vectorEigenToMsg(enu_velocity, twist->twist.linear);
109  twist->twist.angular = baselink_angular_msg;
110 
111  odom->header.stamp = pose->header.stamp;
112  odom->header.frame_id = tf_frame_id;
113  odom->child_frame_id = tf_child_frame_id;
114  tf::vectorEigenToMsg(baselink_linear, odom->twist.twist.linear);
115  odom->twist.twist.angular = baselink_angular_msg;
116  // reasonable defaults for covariance
117  odom->pose.pose = pose->pose;
118  for (int i=0; i< 3; i++) {
119  // linear velocity
120  odom->twist.covariance[i + 6*i] = 1e-4;
121  // angular velocity
122  odom->twist.covariance[(i + 3) + 6*(i + 3)] = 1e-4;
123  // position/ attitude
124  if (i==2) {
125  // z
126  odom->pose.covariance[i + 6*i] = 1e-6;
127  // yaw
128  odom->pose.covariance[(i + 3) + 6*(i + 3)] = 1e-6;
129  } else {
130  // x, y
131  odom->pose.covariance[i + 6*i] = 1e-6;
132  // roll, pitch
133  odom->pose.covariance[(i + 3) + 6*(i + 3)] = 1e-6;
134  }
135  }
136 
137  //--------------- Publish Data ---------------//
138  local_odom.publish(odom);
139  local_position.publish(pose);
140  local_velocity.publish(twist);
141 
142  if (tf_send) {
143  geometry_msgs::TransformStamped transform;
144 
145  transform.header.stamp = pose->header.stamp;
146  transform.header.frame_id = tf_frame_id;
147  transform.child_frame_id = tf_child_frame_id;
148 
149  transform.transform.rotation = enu_orientation_msg;
150  tf::vectorEigenToMsg(enu_position, transform.transform.translation);
151 
152  m_uas->tf2_broadcaster.sendTransform(transform);
153  }
154  if (tf_send_fcu) {
155  //--------------- Report NED->aircraft transform ---------------//
156  geometry_msgs::TransformStamped ned_aircraft_tf;
157 
158  ned_aircraft_tf.header.stamp = pose->header.stamp;
159  ned_aircraft_tf.header.frame_id = "NED";
160  ned_aircraft_tf.child_frame_id = "aircraft";
161 
162  //Don't just report the data from the mavlink message,
163  //actually perform rotations to see if anything is
164  //wrong.
165  auto ned_position = ftf::transform_frame_enu_ned(enu_position);
166  tf::vectorEigenToMsg(ned_position, ned_aircraft_tf.transform.translation);
167 
168  auto ned_orientation = ftf::transform_orientation_enu_ned(
169  ftf::transform_orientation_baselink_aircraft(enu_orientation));
170  tf::quaternionEigenToMsg(ned_orientation,ned_aircraft_tf.transform.rotation);
171  m_uas->tf2_broadcaster.sendTransform(ned_aircraft_tf);
172  }
173  }
