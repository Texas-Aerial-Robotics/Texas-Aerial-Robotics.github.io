54 void enu_2_gym(nav_msgs::Odometry current_pose_enu)

56  float GYM_OFFSET = gymOffset.data;

59  float z = current_pose_enu.pose.pose.position.z;
60  float deg2rad = (M_PI/180);
61  float X = x*cos(GYM_OFFSET*deg2rad) - y*sin(GYM_OFFSET*deg2rad);

64  current_pose.pose.pose.position.x = X;

110 geometry_msgs::PointStamped pixel2metric_facedown(double alt, vector<double> obj_pix, orientation camparamOfFrame)
111 {
112 float theta_x_offset;
113 float theta_y_offset;
114 
115  //use drone attitude to modify camera orientation angles
116  if(camparamOfFrame.frame_id=="camera_link"){theta_x_offset=-current_orientation.roll;
117  theta_y_offset=-current_orientation.pitch;
118 
119  }else if(camparamOfFrame.frame_id=="camera_link2"){theta_x_offset=-current_orientation.pitch;
120  theta_y_offset=current_orientation.roll;
121 
122  }else if(camparamOfFrame.frame_id== "camera_link3"){theta_x_offset=current_orientation.roll;
123  theta_y_offset=current_orientation.pitch;
124  }else if(camparamOfFrame.frame_id== "camera_link4"){theta_x_offset=current_orientation.pitch;
125  theta_y_offset=-current_orientation.roll;
126  }else if(camparamOfFrame.frame_id=="camera_link5"){theta_x_offset=-current_orientation.roll;
127  theta_y_offset=-current_orientation.pitch;}
128 
129 
130 
131  //find pixel of interest in x direction
132  double THETA_Y = camparamOfFrame.pitch*(M_PI/180)+theta_y_offset;
133  double THETA_X = camparamOfFrame.roll*(M_PI/180)+theta_x_offset;
134  double T_x=obj_pix[0];
135  double psi_x;
136  double O_mx;
137 
138  psi_x=2*abs(T_x/PIXELS[0]*PHI_X-PHI_X/2);
139  // find pixel of interest in y direction
140  double T_y=obj_pix[1];
141  double O_my;
142  double psi_y;
143  double hyp_x;
144  double hyp_y;
145  //calculate slice of field of interest
146  psi_y=2*abs(T_y/PIXELS[1]*PHI_Y-PHI_Y/2);
147  double gamma_y;
148  double gamma_x;
149  //double O_mx = r_p[0]/3779.527;
150  if(T_y<PIXELS[1]/2){
151  gamma_y= THETA_Y+psi_y/2;
152  }else{
153  gamma_y= THETA_Y-psi_y/2;
154  }
155  hyp_x=alt/cos(gamma_y);
156  if(T_x>PIXELS[0]/2){
157  gamma_x=THETA_X+psi_x/2;
158  }else{
159  gamma_x=THETA_X-psi_x/2;
160  }
161  hyp_y=alt/cos(gamma_x);
162 
163  O_mx=hyp_x*tan(gamma_x);
164  O_my = alt*tan(gamma_y);
165 
166 
167  ROS_INFO("transformed to x: %f y: %f meters \n", O_mx, O_my);
168 
169 
170  //put transformed point in drone reference frame
171  float GYM_OFFSET = gymOffset.data;
172  float deg2rad = (M_PI/180);
173  float O_mx_drone = O_mx*cos(camparamOfFrame.yaw*deg2rad) - O_my*sin(camparamOfFrame.yaw*deg2rad);
174  float O_my_drone = O_mx*sin(camparamOfFrame.yaw*deg2rad) + O_my*cos(camparamOfFrame.yaw*deg2rad);
175  //ROS_INFO("cam params to x: %f y: %f meters \n", O_mx_drone, O_my_drone);
176  //put point in gym reference frame
177  float X = O_mx_drone*cos(-(current_heading.data-GYM_OFFSET )*deg2rad) - O_my_drone*sin(-(current_heading.data-GYM_OFFSET)*deg2rad);
178  float Y = O_mx_drone*sin(-(current_heading.data-GYM_OFFSET )*deg2rad) + O_my_drone*cos(-(current_heading.data-GYM_OFFSET)*deg2rad);
179 
180  geometry_msgs::PointStamped roombaPose;
181  roombaPose.point.x = X + current_pose.pose.pose.position.x;
182  roombaPose.point.y = Y + current_pose.pose.pose.position.y;
183  roombaPose.point.z = 0;
184  //ROS_INFO("pose gym x: %f y: %f z: %f", current_pose.pose.pose.position.x, current_pose.pose.pose.position.y, current_pose.pose.pose.position.z);
185  ROS_INFO("roombaPose gym x: %f y: %f z: %f", roombaPose.point.x, roombaPose.point.y, roombaPose.point.z);
