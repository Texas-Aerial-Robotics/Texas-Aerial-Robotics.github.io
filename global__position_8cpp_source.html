154  void handle_gps_raw_int(const mavlink::mavlink_message_t *msg, mavlink::common::msg::GPS_RAW_INT &raw_gps)
155  {
156  auto fix = boost::make_shared<sensor_msgs::NavSatFix>();
157 
158  fix->header = m_uas->synchronized_header(child_frame_id, raw_gps.time_usec);
159 
160  fix->status.service = sensor_msgs::NavSatStatus::SERVICE_GPS;
161  if (raw_gps.fix_type > 2)
162  fix->status.status = sensor_msgs::NavSatStatus::STATUS_FIX;
163  else {
164  ROS_WARN_THROTTLE_NAMED(30, "global_position", "GP: No GPS fix");
165  fix->status.status = sensor_msgs::NavSatStatus::STATUS_NO_FIX;
166  }
167 
168  fill_lla(raw_gps, fix);
169 
170  float eph = (raw_gps.eph != UINT16_MAX) ? raw_gps.eph / 1E2F : NAN;
171  float epv = (raw_gps.epv != UINT16_MAX) ? raw_gps.epv / 1E2F : NAN;
172 
173  ftf::EigenMapCovariance3d gps_cov(fix->position_covariance.data());
174 
175  // With mavlink v2.0 use accuracies reported by sensor
176  if (msg->magic == MAVLINK_STX &&
177  raw_gps.h_acc > 0 && raw_gps.v_acc > 0) {
178  gps_cov.diagonal() << std::pow(raw_gps.h_acc / 1E3, 2), std::pow(raw_gps.h_acc / 1E3, 2), std::pow(raw_gps.v_acc / 1E3, 2);
179  fix->position_covariance_type = sensor_msgs::NavSatFix::COVARIANCE_TYPE_DIAGONAL_KNOWN;
180  }
181  // With mavlink v1.0 approximate accuracies by DOP
182  else if (!std::isnan(eph) && !std::isnan(epv)) {
183  gps_cov.diagonal() << std::pow(eph * gps_uere, 2), std::pow(eph * gps_uere, 2), std::pow(epv * gps_uere, 2);
184  fix->position_covariance_type = sensor_msgs::NavSatFix::COVARIANCE_TYPE_APPROXIMATED;
185  }
186  else {
187  fill_unknown_cov(fix);
188  }
189 
190  // store & publish
191  m_uas->update_gps_fix_epts(fix, eph, epv, raw_gps.fix_type, raw_gps.satellites_visible);
192  raw_fix_pub.publish(fix);
193 
194  if (raw_gps.vel != UINT16_MAX &&
195  raw_gps.cog != UINT16_MAX) {
196  double speed = raw_gps.vel / 1E2; // m/s
197  double course = angles::from_degrees(raw_gps.cog / 1E2); // rad
198 
199  auto vel = boost::make_shared<geometry_msgs::TwistStamped>();
200 
201  vel->header.stamp = fix->header.stamp;
202  vel->header.frame_id = frame_id;
203 
204  // From nmea_navsat_driver
205  vel->twist.linear.x = speed * std::sin(course);
206  vel->twist.linear.y = speed * std::cos(course);
207 
208  raw_vel_pub.publish(vel);
209  }
210  }

240  void handle_global_position_int(const mavlink::mavlink_message_t *msg, mavlink::common::msg::GLOBAL_POSITION_INT &gpos)
241  {
242  auto odom = boost::make_shared<nav_msgs::Odometry>();
243  auto fix = boost::make_shared<sensor_msgs::NavSatFix>();
244  auto relative_alt = boost::make_shared<std_msgs::Float64>();
245  auto compass_heading = boost::make_shared<std_msgs::Float64>();
246 
247  auto header = m_uas->synchronized_header(child_frame_id, gpos.time_boot_ms);
248 
249  // Global position fix
250  fix->header = header;
251 
252  fill_lla(gpos, fix);
253 
254  // fill GPS status fields using GPS_RAW data
255  auto raw_fix = m_uas->get_gps_fix();
256  if (raw_fix) {
257  fix->status.service = raw_fix->status.service;
258  fix->status.status = raw_fix->status.status;
259  fix->position_covariance = raw_fix->position_covariance;
260  fix->position_covariance_type = raw_fix->position_covariance_type;
261  }
262  else {
263  // no GPS_RAW_INT -> fix status unknown
264  fix->status.service = sensor_msgs::NavSatStatus::SERVICE_GPS;
265  fix->status.status = sensor_msgs::NavSatStatus::STATUS_NO_FIX;
266 
267  // we don't know covariance
268  fill_unknown_cov(fix);
269  }
270 
271  relative_alt->data = gpos.relative_alt / 1E3; // in meters
272  compass_heading->data = (gpos.hdg != UINT16_MAX) ? gpos.hdg / 1E2 : NAN; // in degrees
273 
287  odom->header.stamp = header.stamp;
288  odom->header.frame_id = frame_id;
289  odom->child_frame_id = child_frame_id;
290 
291  // Linear velocity
292  tf::vectorEigenToMsg(Eigen::Vector3d(gpos.vy, gpos.vx, gpos.vz) / 1E2,
293  odom->twist.twist.linear);
294 
295  // Velocity covariance unknown
296  ftf::EigenMapCovariance6d vel_cov_out(odom->twist.covariance.data());
297  vel_cov_out.fill(0.0);
298  vel_cov_out(0) = -1.0;
299 
300  // Current fix in ECEF
301  Eigen::Vector3d map_point;
302 
303  try {
311  GeographicLib::Geocentric map(GeographicLib::Constants::WGS84_a(),
312  GeographicLib::Constants::WGS84_f());
313 
321  // Current fix to ECEF
322  map.Forward(fix->latitude, fix->longitude, fix->altitude,
323  map_point.x(), map_point.y(), map_point.z());
324 
325  // Set the current fix as the "map" origin if it's not set
326  if (!is_map_init) {
327  map_origin.x() = fix->latitude;
328  map_origin.y() = fix->longitude;
329  map_origin.z() = fix->altitude;
330 
331  ecef_origin = map_point; // Local position is zero
332  is_map_init = true;
333  }
334  }
335  catch (const std::exception& e) {
336  ROS_INFO_STREAM("GP: Caught exception: " << e.what() << std::endl);
337  }
338 
339  // Compute the local coordinates in ECEF
340  local_ecef = map_point - ecef_origin;
341  // Compute the local coordinates in ENU
342  tf::pointEigenToMsg(ftf::transform_frame_ecef_enu(local_ecef, map_origin), odom->pose.pose.position);
343 
348  if (use_relative_alt)
349  odom->pose.pose.position.z = relative_alt->data;
350 
351  odom->pose.pose.orientation = m_uas->get_attitude_orientation_enu();
352 
353  // Use ENU covariance to build XYZRPY covariance
354  ftf::EigenMapConstCovariance3d gps_cov(fix->position_covariance.data());
355  ftf::EigenMapCovariance6d pos_cov_out(odom->pose.covariance.data());
356  pos_cov_out.setZero();
357  pos_cov_out.block<3, 3>(0, 0) = gps_cov;
358  pos_cov_out.block<3, 3>(3, 3).diagonal() <<
359  rot_cov,
360  rot_cov,
361  rot_cov;
362 
363  // publish
364  gp_fix_pub.publish(fix);
365  gp_odom_pub.publish(odom);
366  gp_rel_alt_pub.publish(relative_alt);
367  gp_hdg_pub.publish(compass_heading);
368 
369  // TF
370  if (tf_send) {
371  geometry_msgs::TransformStamped transform;
372 
373  transform.header.stamp = odom->header.stamp;
374  transform.header.frame_id = tf_frame_id;
375  transform.child_frame_id = tf_child_frame_id;
376 
377  // setRotation()
378  transform.transform.rotation = odom->pose.pose.orientation;
379 
380  // setOrigin()
381  transform.transform.translation.x = odom->pose.pose.position.x;
382  transform.transform.translation.y = odom->pose.pose.position.y;
383  transform.transform.translation.z = odom->pose.pose.position.z;
384 
385  m_uas->tf2_broadcaster.sendTransform(transform);
386  }
387  }

389  void handle_lpned_system_global_offset(const mavlink::mavlink_message_t *msg, mavlink::common::msg::LOCAL_POSITION_NED_SYSTEM_GLOBAL_OFFSET &offset)
390  {
391  auto global_offset = boost::make_shared<geometry_msgs::PoseStamped>();
392  global_offset->header = m_uas->synchronized_header(tf_global_frame_id, offset.time_boot_ms);
393 
394  auto enu_position = ftf::transform_frame_ned_enu(Eigen::Vector3d(offset.x, offset.y, offset.z));
395  auto enu_baselink_orientation = ftf::transform_orientation_aircraft_baselink(
396  ftf::transform_orientation_ned_enu(
397  ftf::quaternion_from_rpy(offset.roll, offset.pitch, offset.yaw)));
398 
399  tf::pointEigenToMsg(enu_position, global_offset->pose.position);
400  tf::quaternionEigenToMsg(enu_baselink_orientation, global_offset->pose.orientation);
401 
402  gp_global_offset_pub.publish(global_offset);
403 
404  // TF
405  if (tf_send) {
406  geometry_msgs::TransformStamped transform;
407 
408  transform.header.stamp = global_offset->header.stamp;
409  transform.header.frame_id = tf_global_frame_id;
410  transform.child_frame_id = tf_frame_id;
411 
412  // setRotation()
413  transform.transform.rotation = global_offset->pose.orientation;
414 
415  // setOrigin()
416  transform.transform.translation.x = global_offset->pose.position.x;
417  transform.transform.translation.y = global_offset->pose.position.y;
418  transform.transform.translation.z = global_offset->pose.position.z;
419 
420  m_uas->tf2_broadcaster.sendTransform(transform);
421  }

446  else

448
