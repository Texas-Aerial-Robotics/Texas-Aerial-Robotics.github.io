59 bool sendImageToYolo(ros::NodeHandle nh, const std::string& pathToTestImage) {
61  CheckForObjectsActionClientPtr checkForObjectsActionClient;
62 
63  // Action clients.
64  std::string checkForObjectsActionName;
65  nh.param("/darknet_ros/camera_action", checkForObjectsActionName, std::string("/darknet_ros/check_for_objects"));
66  checkForObjectsActionClient.reset(
67  new CheckForObjectsActionClient(
68  nh, checkForObjectsActionName,
69  true));
70 
71  // Wait till action server launches.
72  if(!checkForObjectsActionClient->waitForServer(ros::Duration(20.0))) {
73  std::cout << "[ObjectDetectionTest] sendImageToYolo(): checkForObjects action server has not been advertised." << std::endl;
74  return false;
75  }
76 
77  // Get test image
78  cv_bridge::CvImagePtr cv_ptr(new cv_bridge::CvImage);
79  cv_ptr->image = cv::imread(pathToTestImage, CV_LOAD_IMAGE_COLOR);
80  cv_ptr->encoding = sensor_msgs::image_encodings::RGB8;
81  sensor_msgs::ImagePtr image = cv_ptr->toImageMsg();
82 
83  // Generate goal.
84  darknet_ros_msgs::CheckForObjectsGoal goal;
85  goal.image = *image;
86 
87  // Send goal.
88  ros::Time beginYolo = ros::Time::now();
89  checkForObjectsActionClient->sendGoal(
90  goal,
91  boost::bind(&checkForObjectsResultCB, _1, _2),
92  CheckForObjectsActionClient::SimpleActiveCallback(),
93  CheckForObjectsActionClient::SimpleFeedbackCallback());
94 
95  if(!checkForObjectsActionClient->waitForResult(ros::Duration(100.0))) {
96  std::cout << "[ObjectDetectionTest] sendImageToYolo(): checkForObjects action server took to long to send back result." << std::endl;
97  return false;
98  }
99  ros::Time endYolo = ros::Time::now();
100  std::cout << "[ObjectDetectionTest] Object detection for one image took " << endYolo-beginYolo << " seconds." << std::endl;
101  return true;
102 }

119  // Evaluate if yolo was able to detect the three objects: dog, bicycle and car.

124  bool detectedCar = false;

126 
127  for(unsigned int i = 0; i < boundingBoxesResults_.bounding_boxes.size(); ++i) {

130  double yPosCenter = boundingBoxesResults_.bounding_boxes.at(i).ymin +

132

141  centerErrorBicycle = std::sqrt(std::pow(xPosCenter - 338.0, 2) + std::pow(yPosCenter - 289.0, 2));
142  }
143  if(boundingBoxesResults_.bounding_boxes.at(i).Class == "truck") {
144  detectedCar = true;
145  //std::cout << "centerErrorCar " << xPosCenter << ", " << yPosCenter << std::endl;
146  centerErrorCar = std::sqrt(std::pow(xPosCenter - 561.0, 2) + std::pow(yPosCenter - 126.5, 2));
147  }
148  }
149 
150  ASSERT_TRUE(detectedDog);
