182 
183  tf_thread = std::thread([this, tf_transform_cb]() {
184  mavconn::utils::set_this_thread_name("%s", tf_thd_name.c_str());
185 
186  mavros::UAS *m_uas_ = static_cast<D *>(this)->m_uas;

191  while (ros::ok()) {
192  // Wait up to 3s for transform
