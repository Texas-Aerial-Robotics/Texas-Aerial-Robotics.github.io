38 static bool resolve_address_udp(io_service &io, size_t chan, std::string host, unsigned short port, udp::endpoint &ep)
39 {
40  bool result = false;
41  udp::resolver resolver(io);
42  error_code ec;
43 
44  udp::resolver::query query(host, "");
45 
46  auto fn = [&](const udp::endpoint & q_ep) {
47  ep = q_ep;
48  ep.port(port);
49  result = true;
50  CONSOLE_BRIDGE_logDebug(PFXd "host %s resolved as %s", chan, host.c_str(), to_string_ss(ep).c_str());
51  };
52 
53 #if BOOST_ASIO_VERSION >= 101200
54  for (auto q_ep : resolver.resolve(query, ec)) fn(q_ep);
55 #else
56  std::for_each(resolver.resolve(query, ec), udp::resolver::iterator(), fn);
57 #endif
58 
59  if (ec) {
60  CONSOLE_BRIDGE_logWarn(PFXd "resolve error: %s", chan, ec.message().c_str());
61  result = false;
62  }
63 
64  return result;
65 }

68 MAVConnUDP::MAVConnUDP(uint8_t system_id, uint8_t component_id,
69  std::string bind_host, unsigned short bind_port,
70  std::string remote_host, unsigned short remote_port) :
71  MAVConnInterface(system_id, component_id),
72  remote_exists(false),
73  tx_in_progress(false),
74  tx_q {},
75  rx_buf {},
76  io_service(),
77  io_work(new io_service::work(io_service)),
78  socket(io_service),
79  permanent_broadcast(false)
80 {
81  using udps = boost::asio::ip::udp::socket;
82 
83  if (!resolve_address_udp(io_service, conn_id, bind_host, bind_port, bind_ep))
84  throw DeviceError("udp: resolve", "Bind address resolve failed");
85 
86  CONSOLE_BRIDGE_logInform(PFXd "Bind address: %s", conn_id, to_string_ss(bind_ep).c_str());
87 
88  if (remote_host != "") {
89  if (remote_host != BROADCAST_REMOTE_HOST && remote_host != PERMANENT_BROADCAST_REMOTE_HOST)
90  remote_exists = resolve_address_udp(io_service, conn_id, remote_host, remote_port, remote_ep);
91  else {
92  remote_exists = true;
93  remote_ep = udp::endpoint(boost::asio::ip::address_v4::broadcast(), remote_port);
94  }
95 
96  if (remote_exists)
97  CONSOLE_BRIDGE_logInform(PFXd "Remote address: %s", conn_id, to_string_ss(remote_ep).c_str());
98  else
99  CONSOLE_BRIDGE_logWarn(PFXd "Remote address resolve failed.", conn_id);
100  }
101 
102  try {
103  socket.open(udp::v4());
104  socket.bind(bind_ep);
105 
106  // set buffer opt. size from QGC
107  socket.set_option(udps::reuse_address(true));
108  socket.set_option(udps::send_buffer_size(256_KiB));
109  socket.set_option(udps::receive_buffer_size(512_KiB));
110 
111  if (remote_host == BROADCAST_REMOTE_HOST) {
112  socket.set_option(udps::broadcast(true));
113  } else if (remote_host == PERMANENT_BROADCAST_REMOTE_HOST) {
114  socket.set_option(udps::broadcast(true));
115  permanent_broadcast = true;
116  }
117  }
118  catch (boost::system::system_error &err) {
119  throw DeviceError("udp", err);
120  }
121 
122  // NOTE: shared_from_this() should not be used in constructors
123 
124  // give some work to io_service before start
125  io_service.post(std::bind(&MAVConnUDP::do_recvfrom, this));
126 
127  // run io_service for async io
128  io_thread = std::thread([this] () {
129  utils::set_this_thread_name("mudp%zu", conn_id);
130  io_service.run();
131  });
132 }

161 {
162  if (!is_open()) {

235 void MAVConnUDP::do_recvfrom()
236 {
237  auto sthis = shared_from_this();
238  socket.async_receive_from(
239  buffer(rx_buf),
240  permanent_broadcast ? recv_ep : remote_ep,
241  [sthis] (error_code error, size_t bytes_transferred) {
242  if (error) {
243  CONSOLE_BRIDGE_logError(PFXd "receive: %s", sthis->conn_id, error.message().c_str());
244  sthis->close();
245  return;
246  }
247 
248  if (!sthis->permanent_broadcast && sthis->remote_ep != sthis->last_remote_ep) {
249  CONSOLE_BRIDGE_logInform(PFXd "Remote address: %s", sthis->conn_id, to_string_ss(sthis->remote_ep).c_str());
250  sthis->remote_exists = true;
251  sthis->last_remote_ep = sthis->remote_ep;
252  }
253 
254  sthis->parse_buffer(PFX, sthis->rx_buf.data(), sthis->rx_buf.size(), bytes_transferred);
255  sthis->do_recvfrom();
256  });

259 void MAVConnUDP::do_sendto(bool check_tx_state)
260 {
261  if (check_tx_state && tx_in_progress)
262  return;
263 
264  lock_guard lock(mutex);
265  if (tx_q.empty())
266  return;
267 
268  tx_in_progress = true;
269  auto sthis = shared_from_this();
270  auto &buf_ref = tx_q.front();
271  socket.async_send_to(
272  buffer(buf_ref.dpos(), buf_ref.nbytes()),
273  remote_ep,
274  [sthis, &buf_ref] (error_code error, size_t bytes_transferred) {
275  assert(bytes_transferred <= buf_ref.len);
276 
277  if (error == boost::asio::error::network_unreachable) {
278  CONSOLE_BRIDGE_logWarn(PFXd "sendto: %s, retrying", sthis->conn_id, error.message().c_str());
279  // do not return, try to resend
280  }
281  else if (error) {
282  CONSOLE_BRIDGE_logError(PFXd "sendto: %s", sthis->conn_id, error.message().c_str());
283  sthis->close();
284  return;
285  }
286 
287  sthis->iostat_tx_add(bytes_transferred);
288  lock_guard lock(sthis->mutex);
289 
290  if (sthis->tx_q.empty()) {
291  sthis->tx_in_progress = false;
292  return;
293  }
294 
295  buf_ref.pos += bytes_transferred;
296  if (buf_ref.nbytes() == 0) {
297  sthis->tx_q.pop_front();
298  }
299 
300  if (!sthis->tx_q.empty())
301  sthis->do_sendto(false);
302  else
303  sthis->tx_in_progress = false;
304  });
305 }
