591  def _rewind(self):
592  '''rewind to start of log'''
593  DFReader._rewind(self)
594  self.offset = 0
595  self.remaining = self.data_len

597  def _parse_next(self):
598  '''read one message, returning it as an object'''
599  if self.data_len - self.offset < 3:
600  return None
601 
602  hdr = self.data[self.offset:self.offset+3]
603  skip_bytes = 0
604  skip_type = None
605  # skip over bad messages
606  while (u_ord(hdr[0]) != self.HEAD1 or u_ord(hdr[1]) != self.HEAD2 or
607  u_ord(hdr[2]) not in self.formats):
608  if skip_type is None:
609  skip_type = (u_ord(hdr[0]), u_ord(hdr[1]), u_ord(hdr[2]))
610  skip_start = self.offset
611  skip_bytes += 1
612  self.offset += 1
613  if self.data_len - self.offset < 3:
614  return None
615  hdr = self.data[self.offset:self.offset+3]
616  msg_type = u_ord(hdr[2])
617  if skip_bytes != 0:
618  if self.remaining < 528:
619  return None
620  print("Skipped %u bad bytes in log at offset %u, type=%s" %
621  (skip_bytes, skip_start, skip_type), file=sys.stderr)
622  self.remaining -= skip_bytes
623 
624  self.offset += 3
625  self.remaining -= 3
626 
627  if msg_type not in self.formats:
628  if self.verbose:
629  print("unknown message type %02x" % msg_type, file=sys.stderr)
630  raise Exception("Unknown message type %02x" % msg_type)
631  fmt = self.formats[msg_type]
632  if self.remaining < fmt.len-3:
633  # out of data - can often happen half way through a message
634  if self.verbose:
635  print("out of data", file=sys.stderr)
636  return None
637  body = self.data[self.offset:self.offset+(fmt.len-3)]
638  elements = None
639  try:
640  elements = list(struct.unpack(fmt.msg_struct, body))
641  except Exception:
642  if self.remaining < 528:
643  # we can have garbage at the end of an APM2 log
644  return None
645  # we should also cope with other corruption; logs
646  # transfered via DataFlash_MAVLink may have blocks of 0s
647  # in them, for example
